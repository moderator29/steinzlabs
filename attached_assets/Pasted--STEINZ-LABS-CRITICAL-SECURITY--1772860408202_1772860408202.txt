SECURITY FIX #1: AUTHENTICATION & SESSION MANAGEMENT
═══════════════════════════════════════════════════════════════════

1. SECURE JWT TOKEN HANDLING:

// lib/auth.ts - UPDATE THIS FILE
import { cookies } from 'next/headers';
import jwt from 'jsonwebtoken';

const JWT_SECRET = process.env.JWT_SECRET!; // Add to .env
const JWT_EXPIRY = '7d';

export async function createSession(userId: string) {
  const token = jwt.sign(
    { 
      userId, 
      iat: Date.now(),
      exp: Date.now() + (7 * 24 * 60 * 60 * 1000) // 7 days
    },
    JWT_SECRET,
    { algorithm: 'HS256' }
  );
  
  // Set httpOnly cookie (NOT accessible via JavaScript)
  cookies().set('auth_token', token, {
    httpOnly: true,      // Prevents XSS attacks
    secure: true,        // HTTPS only
    sameSite: 'strict',  // Prevents CSRF
    maxAge: 7 * 24 * 60 * 60, // 7 days
    path: '/',
  });
  
  return token;
}

export async function verifySession(): Promise<string | null> {
  try {
    const token = cookies().get('auth_token')?.value;
    
    if (!token) return null;
    
    const decoded = jwt.verify(token, JWT_SECRET) as { userId: string };
    return decoded.userId;
  } catch (error) {
    // Token invalid or expired
    cookies().delete('auth_token');
    return null;
  }
}

export async function destroySession() {
  cookies().delete('auth_token');
}

// lib/rateLimit.ts - NEW FILE
import { LRUCache } from 'lru-cache';

const rateLimitCache = new LRUCache({
  max: 500,
  ttl: 60000, // 1 minute
});

export function rateLimit(identifier: string, limit: number = 5): boolean {
  const attempts = (rateLimitCache.get(identifier) as number) || 0;
  
  if (attempts >= limit) {
    return false; // Rate limited
  }
  
  rateLimitCache.set(identifier, attempts + 1);
  return true; // Allowed
}

// Prevent wallet address spoofing
export async function verifyWalletOwnership(address: string) {
  // Generate random challenge
  const challenge = `Sign this to verify you own ${address}\n\nNonce: ${Date.now()}`;
  
  // Request signature from wallet
  const provider = new ethers.BrowserProvider(window.ethereum);
  const signer = await provider.getSigner();
  const signature = await signer.signMessage(challenge);
  
  // Verify signature on backend
  const recoveredAddress = ethers.verifyMessage(challenge, signature);
  
  if (recoveredAddress.toLowerCase() !== address.toLowerCase()) {
    throw new Error('Wallet ownership verification failed');
  }
  
  return true;
}

export async function executeTrade(params: TradeParams) {
  // Step 1: Simulate transaction FIRST
  const simulation = await simulateTransaction(params);
  
  if (!simulation.success) {
    throw new Error(`Transaction would fail: ${simulation.error}`);
  }
  
  if (simulation.warnings.length > 0) {
    // Return warnings to user for confirmation
    return {
      needsConfirmation: true,
      warnings: simulation.warnings,
      simulation,
    };
  }
  
  // Step 2: Get user confirmation
  // (Frontend handles this)
  
  // Step 3: Build transaction with safety checks
  const tx = await buildTransaction(params);
  
  // Verify slippage protection
  if (tx.slippage > params.maxSlippage) {
    throw new Error('Slippage too high');
  }
  
  // Verify amounts
  if (tx.amountIn !== params.amountIn) {
    throw new Error('Amount mismatch');
  }
  
  return tx; // User signs in wallet
}
