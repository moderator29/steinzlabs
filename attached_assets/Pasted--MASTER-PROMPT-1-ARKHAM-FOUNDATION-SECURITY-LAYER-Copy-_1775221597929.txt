import { 
  ArkhamEntity,

class ArkhamAPI {
  private apiKey = process.env.ARKHAM_API_KEY!;
  private baseUrl = 'https://api.arkm.com';
  private cache = new Map<string, { data: any; timestamp: number }>();
  private cacheDuration = 5 * 60 * 1000; // 5 minutes

  // Helper: Make authenticated request
  private async fetch(endpoint: string, options?: RequestInit) {
    const url = `${this.baseUrl}${endpoint}`;
    
    try {
      const response = await fetch(url, {
        ...options,
        headers: {
          'API-Key': this.apiKey,
          'Content-Type': 'application/json',
          ...options?.headers,
        },
      });

      if (!response.ok) {
        const error = await response.text();
        throw new Error(`Arkham API error (${response.status}): ${error}`);
      }

      return response.json();
    } catch (error) {
      console.error('Arkham API request failed:', error);
      throw error;
    }
  }

  // Helper: Cache management
  private getCached(key: string): any | null {
    const cached = this.cache.get(key);
    if (!cached) return null;
    
    const isExpired = Date.now() - cached.timestamp > this.cacheDuration;
    if (isExpired) {
      this.cache.delete(key);
      return null;
    }
    
    return cached.data;
  }

  private setCache(key: string, data: any): void {
    this.cache.set(key, { data, timestamp: Date.now() });
  }

  // ===================================================================
  // ENDPOINT 1: Get Address Intelligence
  // ===================================================================
  async getAddressIntel(address: string, chain?: string): Promise<ArkhamAddress> {
    const cacheKey = `address:${address}:${chain || 'all'}`;
    const cached = this.getCached(cacheKey);
    if (cached) return cached;

    const endpoint = chain 
      ? `/intelligence/address/${address}?chain=${chain}`
      : `/intelligence/address/${address}`;

    const data = await this.fetch(endpoint);
    this.setCache(cacheKey, data);
    return data;
  }

  // ===================================================================
  // ENDPOINT 2: Get Entity Details
  // ===================================================================
  async getEntity(entityId: string): Promise<ArkhamEntity> {
    const cacheKey = `entity:${entityId}`;
    const cached = this.getCached(cacheKey);
    if (cached) return cached;

    const data = await this.fetch(`/intelligence/entity/${entityId}`);
    this.setCache(cacheKey, data);
    return data;
  }

  // ===================================================================
  // ENDPOINT 3: Get Token Holders (with entity enrichment)
  // ===================================================================
  async getTokenHolders(
    tokenAddress: string, 
    limit: number = 20,
    chain?: string
  ): Promise<ArkhamHolder[]> {
    const cacheKey = `holders:${tokenAddress}:${limit}:${chain || 'all'}`;
    const cached = this.getCached(cacheKey);
    if (cached) return cached;

    const endpoint = chain
      ? `/token/${tokenAddress}/holders?limit=${limit}&chain=${chain}`
      : `/token/${tokenAddress}/holders?limit=${limit}`;

    const data = await this.fetch(endpoint);
    
    // Enrich each holder with entity data
    const enriched = await Promise.all(
      (data.holders || []).map(async (holder: any) => {
        try {
          const addressIntel = await this.getAddressIntel(holder.address);
          return {
            ...holder,
            entity: addressIntel.arkhamEntity,
            labels: addressIntel.labels,
          };
        } catch (error) {
          // If enrichment fails, return basic holder data
          return holder;
        }
      })
    );

    this.setCache(cacheKey, enriched);
    return enriched;
  }

  // ===================================================================
  // ENDPOINT 4: Get Wallet Network Connections
  // ===================================================================
  async getWalletConnections(
    address: string,
    limit: number = 50
  ): Promise<ArkhamConnection[]> {
    const cacheKey = `connections:${address}:${limit}`;
    const cached = this.getCached(cacheKey);
    if (cached) return cached;

    const data = await this.fetch(
      `/intelligence/address/${address}/connections?limit=${limit}`
    );
    
    this.setCache(cacheKey, data.connections || []);
    return data.connections || [];
  }

  // ===================================================================
  // ENDPOINT 5: Search by Label (get all wallets with label)
  // ===================================================================
  async getByLabel(label: string): Promise<ArkhamAddress[]> {
    const cacheKey = `label:${label}`;
    const cached = this.getCached(cacheKey);
    if (cached) return cached;

    const data = await this.fetch(`/intelligence/labels/${label}`);
    this.setCache(cacheKey, data.addresses || []);
    return data.addresses || [];
  }

  // ===================================================================
  // ENDPOINT 6: Get Scammer Database
  // ===================================================================
  async getScammerDatabase(): Promise<ArkhamAddress[]> {
    return this.getByLabel('scammer');
  }

  // ===================================================================
  // ENDPOINT 7: Get Transaction Details
  // ===================================================================
  async getTransaction(txHash: string): Promise<ArkhamTransaction> {
    const cacheKey = `tx:${txHash}`;
    const cached = this.getCached(cacheKey);
    if (cached) return cached;

    const data = await this.fetch(`/tx/${txHash}`);
    this.setCache(cacheKey, data);
    return data;
  }

  // ===================================================================
  // ENDPOINT 8: Get Address Transfers (recent activity)
  // ===================================================================
  async getAddressTransfers(
    address: string,
    limit: number = 100,
    chain?: string
  ): Promise<ArkhamTransaction[]> {
    const params = new URLSearchParams({
      limit: limit.toString(),
      ...(chain && { chain }),
    });

    const data = await this.fetch(
      `/transfers/address/${address}?${params.toString()}`
    );
    
    return data.transfers || [];
  }

  // ===================================================================
  // ENDPOINT 9: Get Entity Portfolio
  // ===================================================================
  async getEntityPortfolio(entityId: string): Promise<EntityPortfolio> {
    const cacheKey = `portfolio:${entityId}`;
    const cached = this.getCached(cacheKey);
    if (cached) return cached;

    const data = await this.fetch(`/portfolio/entity/${entityId}`);
    this.setCache(cacheKey, data);
    return data;
  }

  // ===================================================================
  // ENDPOINT 10: Get Entity Performance Metrics
  // ===================================================================
  async getEntityPerformance(entityId: string): Promise<EntityPerformance> {
    const cacheKey = `performance:${entityId}`;
    const cached = this.getCached(cacheKey);
    if (cached) return cached;

    const data = await this.fetch(`/intelligence/entity/${entityId}/performance`);
    this.setCache(cacheKey, data);
    return data;
  }

  // HELPER: Check if address is scammer
  // ===================================================================
  async isScammer(address: string): Promise<boolean> {
    try {
      const intel = await this.getAddressIntel(address);
      return (
        intel.labels?.includes('scammer') ||
        intel.labels?.includes('rug_puller') ||
        intel.labels?.includes('phishing') ||
        !!intel.scamHistory
      );
    } catch (error) {
      console.error('Error checking if address is scammer:', error);
      return false;
    }
  }

  // HELPER: Check if address is verified entity
  // ===================================================================
  async isVerifiedEntity(address: string): Promise<boolean> {
    try {
      const intel = await this.getAddressIntel(address);
      return intel.arkhamEntity?.verified === true;
    } catch (error) {
      return false;
    }
  }

  // ===================================================================
  // HELPER: Get entity by wallet address
  // ===================================================================
  async getEntityByAddress(address: string): Promise<ArkhamEntity | null> {
    try {
      const intel = await this.getAddressIntel(address);

      return data.entities || [];

      return [];
    }
  }

import { ScanResult, ScammerInfo } from './types';

class ShadowGuardian {
  
  // ===================================================================
  // MAIN: Scan trade before execution
  // ===================================================================
  async scanTrade(
    tokenAddress: string,
    amount: number,
    userWallet?: string
  ): Promise<ScanResult> {
    
    console.log('🛡️ Shadow Guardian: Starting scan...', { tokenAddress, amount });

    try {
      // STEP 1: Get token holders from Arkham
      const holders = await arkhamAPI.getTokenHolders(tokenAddress, 20);
      
      if (!holders || holders.length === 0) {
        return {
          allowed: false,
          blocked: true,
          riskScore: 10,
          reason: 'UNABLE_TO_VERIFY',
          recommendation: 'BLOCKED',
          message: '⚠️ Cannot verify token holders. Trade blocked for safety.',
        };
      }

      // STEP 2: Check for scammers in top holders
      const scammers: ScammerInfo[] = [];
      const verifiedHolders: string[] = [];

      for (const holder of holders) {
        // Check if this holder is a known scammer
        const isScammer = holder.labels?.includes('scammer') || 
                         holder.labels?.includes('rug_puller');

        if (isScammer) {
          const addressIntel = await arkhamAPI.getAddressIntel(holder.address);
          
          scammers.push({
            address: holder.address,
            name: addressIntel.arkhamEntity?.name || 'Unknown Scammer',
            rugPulls: addressIntel.scamHistory?.totalRugs || 0,
            totalStolen: addressIntel.scamHistory?.totalStolen || '$0',
            victims: addressIntel.scamHistory?.victims || 0,
            lastScam: addressIntel.scamHistory?.lastScam || 'Unknown',
          });
        }

        // Track verified entities
        if (holder.entity?.verified) {
          verifiedHolders.push(holder.entity.name);
        }
      }

      // STEP 3: If scammers detected, BLOCK immediately
      if (scammers.length > 0) {
        return {
          allowed: false,
          blocked: true,
          riskScore: 10,
          reason: 'KNOWN_SCAMMER_DETECTED',
          scammers,
          verifiedHolders,
          recommendation: 'BLOCKED',
          message: `🚨 TRADE BLOCKED: ${scammers.length} known scammer(s) detected in top holders`,
        };
      }

      // STEP 4: Network analysis - check for mixer connections
      const topHolder = holders[0];
      const connections = await arkhamAPI.getWalletConnections(topHolder.address, 50);
      
      const suspiciousConnections = connections.filter(conn =>
        conn.labels?.includes('mixer') ||
        conn.labels?.includes('tornado_cash') ||
        conn.labels?.includes('mule')
      );

      if (suspiciousConnections.length > 10) {
        return {
          allowed: false,
          blocked: true,
          riskScore: 9,
          reason: 'SUSPICIOUS_NETWORK',
          suspiciousConnections: suspiciousConnections.length,
          verifiedHolders,
          recommendation: 'BLOCKED',
          message: `⚠️ TRADE BLOCKED: Top holder connected to ${suspiciousConnections.length} mixer/mule wallets`,
        };
      }

      // STEP 5: AI risk analysis
      const aiAnalysis = await this.getAIRiskAssessment({
        tokenAddress,
        holders: holders.slice(0, 5),
        verifiedCount: verifiedHolders.length,
        suspiciousCount: suspiciousConnections.length,
      });

      const aiRisk = this.extractRiskScore(aiAnalysis);

      // STEP 6: Final decision based on AI risk
      if (aiRisk >= 7) {
        return {
          allowed: false,
          blocked: true,
          riskScore: aiRisk,
          reason: 'AI_HIGH_RISK',
          aiAnalysis,
          verifiedHolders,
          suspiciousConnections: suspiciousConnections.length,
          recommendation: 'BLOCKED',
          message: `⚠️ TRADE BLOCKED: AI detected high risk (${aiRisk}/10)`,
        };
      }

      // STEP 7: All checks passed - allow trade
      return {
        allowed: true,
        blocked: false,
        riskScore: aiRisk,
        verifiedHolders,
        suspiciousConnections: suspiciousConnections.length,
        aiAnalysis,
        recommendation: aiRisk < 4 ? 'SAFE' : 'CAUTION',
        message: `✅ SAFE TO TRADE: No threats detected (Risk: ${aiRisk}/10)`,
      };

    } catch (error) {
      console.error('Shadow Guardian scan failed:', error);
      
      // On error, block for safety
      return {
        allowed: false,
        blocked: true,
        riskScore: 10,
        reason: 'SCAN_ERROR',
        recommendation: 'BLOCKED',
        message: '⚠️ Scan failed. Trade blocked for safety.',

  private async getAIRiskAssessment(data: any): Promise<string> {

Token: ${data.tokenAddress}
Top 5 Holders: ${JSON.stringify(data.holders, null, 2)}
Verified Entities: ${data.verifiedCount}
Suspicious Connections: ${data.suspiciousCount}

Analyze this token and provide:
1. Risk score from 0-10 (0 = completely safe, 10 = extreme danger)
2. Brief explanation of your assessment
3. Recommendation (SAFE / CAUTION / BLOCKED)

Format your response as:
RISK_SCORE: [number]
ANALYSIS: [your analysis]`;

    try {
      const analysis = await anthropicAPI.analyze(prompt);
      return analysis;
    } catch (error) {
      console.error('AI risk assessment failed:', error);
      return 'RISK_SCORE: 7\nANALYSIS: Unable to complete AI analysis. Proceeding with caution.';
    }
  }

  // ===================================================================
  // HELPER: Extract risk score from AI response
  // ===================================================================
  private extractRiskScore(analysis: string): number {
    const match = analysis.match(/RISK_SCORE:\s*(\d+)/i);
    if (match && match[1]) {
      const score = parseInt(match[1]);
      return Math.min(Math.max(score, 0), 10); // Clamp 0-10
    }
    return 5; // Default to medium risk if can't parse
  }

  // ===================================================================
  // QUICK SCAN: For search results (faster, less detailed)
  // ===================================================================

      );
      if (hasScammer) riskScore += 10;

      // Check for verified entities (reduces risk)
      const verifiedCount = holders.filter(h => h.entity?.verified).length;
      riskScore -= verifiedCount * 0.5;

      // Clamp to 0-10
      return Math.min(Math.max(riskScore, 0), 10);

    } catch (error) {
      return 5; // Default medium risk on error
    }
  }

import { arkhamAPI } from '../arkham/api';

export async function getWalletReputation(
  walletAddress: string
): Promise<WalletReputation> {
  
  try {
    const intel = await arkhamAPI.getAddressIntel(walletAddress);

    // Check if wallet is flagged as malicious
    const isFlagged =
      intel.labels?.includes('scammer') ||
      intel.labels?.includes('rug_puller') ||
      intel.labels?.includes('phishing') ||
      intel.labels?.includes('hacked') ||
      !!intel.scamHistory;

    if (isFlagged) {
      return {
        score: 0,
        reputation: 'DANGEROUS',
        verified: false,
        entity: intel.arkhamEntity,
        warnings: [
          '🚨 This wallet is flagged as malicious',
          '⚠️ Previous scam activity detected',
          '⚠️ Connected to known bad actors',
        ],
        benefits: [],
        allowAccess: false, // Block from platform
      };
    }

    // Check if verified entity
    if (intel.arkhamEntity?.verified) {
      return {
        score: 100,
        reputation: 'VERIFIED',
        verified: true,
        entity: intel.arkhamEntity,
        warnings: [],
        benefits: [
          '✅ Verified institutional wallet',
          '✅ Trusted entity',
          '✅ Full platform access',
        ],
        allowAccess: true,
      };
    }

    // Check for suspicious labels
    const isSuspicious =
      intel.labels?.includes('mixer') ||
      intel.labels?.includes('high_risk');

    if (isSuspicious) {
      return {
        score: 30,
        reputation: 'SUSPICIOUS',
        verified: false,
        entity: null,
        warnings: [
          '⚠️ Wallet has suspicious activity',
          '⚠️ Connected to mixers or high-risk addresses',
        ],
        benefits: [],
        allowAccess: true, // Allow but monitor
      };
    }

    // Unknown wallet - neutral score
    return {
      score: 50,
      reputation: 'UNKNOWN',
      verified: false,
      entity: null,
      warnings: [],
      benefits: [],
      allowAccess: true,
    };

  } catch (error) {
    console.error('Wallet reputation check failed:', error);
    
    // On error, allow but with low score
    return {
      score: 40,
      reputation: 'UNKNOWN',
      verified: false,
      entity: null,
      warnings: ['⚠️ Unable to verify wallet reputation'],
      benefits: [],
      allowAccess: true,
    };
  }
}

      const response = await fetch('https://api.anthropic.com/v1/messages', {

        headers: {
          'Content-Type': 'application/json',
          'x-api-key': this.apiKey,
          'anthropic-version': '2023-06-01',
        },
        body: JSON.stringify({
          model: 'claude-sonnet-4-20250514',
          max_tokens: 1000,
          messages: [
            {
              role: 'user',
              content: prompt,
            },
          ],
        }),
      });

      if (!response.ok) {
        throw new Error(`Anthropic API error: ${response.statusText}`);
      }

      const data = await response.json();
      return data.content[0].text;
    } catch (error) {
      console.error('Anthropic API request failed:', error);
      throw error;
    }
  }
}

import { NextRequest, NextResponse } from 'next/server';

export async function POST(request: NextRequest) {
  try {
    const { tokenAddress, amount, userWallet } = await request.json();

    if (!tokenAddress) {
      return NextResponse.json(
        { error: 'tokenAddress is required' },
        { status: 400 }
      );
    }

    // Run Shadow Guardian scan
    const scanResult = await shadowGuardian.scanTrade(
      tokenAddress,
      amount,
      userWallet
    );

    // Save scan to database if user is logged in
    if (userWallet) {
      const user = await getUserByWallet(userWallet);
      if (user) {
        await saveScanResult({
          userId: user.id,
          tokenAddress,
          scanResult,
          allowed: scanResult.allowed,
          blocked: scanResult.blocked,
          riskScore: scanResult.riskScore,
          reason: scanResult.reason,
        });
      }
    }

    return NextResponse.json(scanResult);
  } catch (error: any) {
    console.error('Shadow Guardian scan failed:', error);
    return NextResponse.json(
      { error: error.message || 'Scan failed' },
      { status: 500 }
    );
  }
}
