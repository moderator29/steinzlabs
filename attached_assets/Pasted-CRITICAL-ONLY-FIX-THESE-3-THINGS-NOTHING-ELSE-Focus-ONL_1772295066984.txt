/**

 * Connects MetaMask (Ethereum) or Phantom (Solana)

declare global {
  interface Window {
    ethereum?: any;
    solana?: any;
  }
}

export async function connectWallet(): Promise<{
  address: string;
  provider: 'metamask' | 'phantom';
  chainId?: string;
}> {
  // Try MetaMask first
  if (typeof window.ethereum !== 'undefined') {
    try {
      console.log('Connecting to MetaMask...');
      
      const accounts = await window.ethereum.request({
        method: 'eth_requestAccounts'
      });

      const chainId = await window.ethereum.request({
        method: 'eth_chainId'
      });

      console.log('MetaMask connected:', accounts[0]);

      return {
        address: accounts[0],
        provider: 'metamask',
        chainId
      };
    } catch (error: any) {
      console.error('MetaMask error:', error);
      throw new Error(`MetaMask: ${error.message}`);
    }
  }

  // Try Phantom if MetaMask not available
  if (typeof window.solana !== 'undefined') {
    try {
      console.log('Connecting to Phantom...');
      
      const response = await window.solana.connect();
      const address = response.publicKey.toString();

      console.log('Phantom connected:', address);

      return {
        address,
        provider: 'phantom'
      };
    } catch (error: any) {
      console.error('Phantom error:', error);
      throw new Error(`Phantom: ${error.message}`);
    }
  }

  throw new Error('No wallet found. Please install MetaMask or Phantom.');
}

export function disconnectWallet() {
  if (window.solana?.disconnect) {
    window.solana.disconnect();
  }
  localStorage.removeItem('wallet_address');
  localStorage.removeItem('wallet_provider');
}

async function generateAISummary(eventDescription: string, eventType: string): Promise<{
  text: string;
  sentiment: 'BULLISH' | 'BEARISH' | 'NEUTRAL';
}> {
  try {
    const message = await anthropic.messages.create({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 200,
      messages: [{
        role: 'user',
        content: `Analyze this on-chain event and provide:
1. A clear 1-2 sentence summary
2. Sentiment classification (BULLISH, BEARISH, or NEUTRAL)

Event: ${eventDescription}
Type: ${eventType}

Format your response as:
[SENTIMENT]
Summary text here`
      }]
    });

    const response = message.content[0].type === 'text' ? message.content[0].text : '';
    
    const lines = response.split('\n').filter(l => l.trim());
    const sentimentLine = lines[0].replace('[', '').replace(']', '').trim();
    const summaryText = lines.slice(1).join(' ').trim();

    let sentiment: 'BULLISH' | 'BEARISH' | 'NEUTRAL' = 'NEUTRAL';
    if (sentimentLine.includes('BULLISH')) sentiment = 'BULLISH';
    else if (sentimentLine.includes('BEARISH')) sentiment = 'BEARISH';

    return {
      text: summaryText || eventDescription,
      sentiment
    };

  } catch (error) {
    console.error('AI summary error:', error);
    return {
      text: eventDescription,
      sentiment: 'NEUTRAL'
    };
  }
}
