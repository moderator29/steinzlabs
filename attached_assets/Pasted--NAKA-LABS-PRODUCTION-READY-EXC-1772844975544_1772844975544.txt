Example Queries to Implement:

Whale Wallet Discovery:
SELECT 
  address,
  COUNT(*) as trade_count,
  SUM(amount_usd) as total_volume,
  AVG(profit_usd) as avg_profit
FROM ethereum.trades
WHERE amount_usd > 100000
  AND block_time > NOW() - INTERVAL '30 days'
GROUP BY address
HAVING AVG(profit_usd) > 0
ORDER BY avg_profit DESC
LIMIT 100

Token Holder Analysis:
SELECT 
  token_address,
  COUNT(DISTINCT holder_address) as unique_holders,
  AVG(balance_usd) as avg_holding,
  PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY balance_usd) as median_holding
FROM token_balances
WHERE token_address = '{{token_address}}'
  AND balance > 0
GROUP BY token_address

Smart Money Tracking:
SELECT 
  buyer_address,
  token_bought,
  buy_timestamp,
  buy_price,
  current_price,
  (current_price - buy_price) / buy_price * 100 as roi_percentage
FROM trades
WHERE buyer_address IN (SELECT address FROM profitable_wallets)
  AND buy_timestamp > NOW() - INTERVAL '7 days'
ORDER BY roi_percentage DESC

class AppError extends Error {
  constructor(
    message: string,
    public code: string,
    public severity: 'low' | 'medium' | 'high' | 'critical',
    public userMessage?: string,
    public metadata?: Record<string, any>
  ) {
    super(message);
    this.name = 'AppError';
  }
}

function handleError(error: unknown) {
  if (error instanceof AppError) {
    // Known error
    logError(error);
    showUserNotification(error.userMessage || 'Something went wrong');
    
    if (error.severity === 'critical') {
      reportToSentry(error);
    }
  } else {
    // Unknown error
    console.error('Unexpected error:', error);
    reportToSentry(error);
    showUserNotification('An unexpected error occurred. Our team has been notified.');
  }
}

Structured Logging:
interface LogEntry {
  timestamp: number;
  level: 'debug' | 'info' | 'warn' | 'error';
  message: string;
  context?: Record<string, any>;
  userId?: string;
  sessionId: string;
}

class Logger {
  private static instance: Logger;
  
  private constructor() {}
  
  static getInstance() {
    if (!Logger.instance) {
      Logger.instance = new Logger();
    }
    return Logger.instance;
  }
  
  info(message: string, context?: Record<string, any>) {
    this.log('info', message, context);
  }
  
  error(message: string, error: Error, context?: Record<string, any>) {
    this.log('error', message, { ...context, error: error.message, stack: error.stack });
  }
  
  private log(level: LogEntry['level'], message: string, context?: Record<string, any>) {
    const entry: LogEntry = {
      timestamp: Date.now(),
      level,
      message,
      context,
      sessionId: getSessionId(),
      userId: getCurrentUserId(),
    };
    
    // Log to console in development
    if (process.env.NODE_ENV === 'development') {
      console[level](message, context);
    }
    
    // Send to logging service in production
    if (process.env.NODE_ENV === 'production') {
      sendToLoggingService(entry);
    }
  }
}
