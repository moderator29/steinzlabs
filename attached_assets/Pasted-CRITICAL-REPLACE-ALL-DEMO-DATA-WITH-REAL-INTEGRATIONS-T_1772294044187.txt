/**

 * Connects to MetaMask (Ethereum) and Phantom (Solana)

export async function connectMetaMask() {
  if (typeof window.ethereum === 'undefined') {
    throw new Error('MetaMask not installed');
  }

  try {
    const accounts = await window.ethereum.request({
      method: 'eth_requestAccounts'
    });

    const address = accounts[0];
    
    // Get chain ID
    const chainId = await window.ethereum.request({
      method: 'eth_chainId'
    });

    return {
      address,
      chainId,
      provider: 'metamask'
    };
  } catch (error: any) {
    throw new Error(error.message);
  }
}

export async function connectPhantom() {
  if (typeof window.solana === 'undefined') {
    throw new Error('Phantom not installed');
  }

  try {
    const response = await window.solana.connect();
    const address = response.publicKey.toString();

    return {
      address,
      provider: 'phantom'
    };
  } catch (error: any) {
    throw new Error(error.message);
  }
}

import { Network, Alchemy } from 'alchemy-sdk';

  apiKey: process.env.ALCHEMY_API_KEY,
  network: Network.ETH_MAINNET,
});

export async function POST(request: Request) {
  try {
    const { address, provider } = await request.json();

    if (provider === 'phantom') {
      // Solana portfolio via Helius
      const response = await fetch(
        `https://api.helius.xyz/v0/addresses/${address}/balances?api-key=${process.env.HELIUS_API_KEY}`
      );
      const data = await response.json();
      
      return NextResponse.json({
        totalValue: data.nativeBalance?.lamports / 1e9 * 150, // SOL price approx
        tokens: data.tokens || [],
        change24h: 0
      });
    }

    // Ethereum via Alchemy
    const balance = await alchemyEth.core.getBalance(address);
    const ethBalance = parseInt(balance.toString()) / 1e18;
    
    const tokens = await alchemyEth.core.getTokenBalances(address);
    
    const tokenData = await Promise.all(
      tokens.tokenBalances.slice(0, 20).map(async (token) => {
        if (token.tokenBalance === '0x0') return null;
        
        const metadata = await alchemyEth.core.getTokenMetadata(token.contractAddress);
        const balance = parseInt(token.tokenBalance!) / Math.pow(10, metadata.decimals || 18);
        
        return {
          symbol: metadata.symbol,
          name: metadata.name,
          balance,
          valueUsd: 0,
          logo: metadata.logo
        };
      })
    );

    const validTokens = tokenData.filter(t => t !== null);

    return NextResponse.json({
      totalValue: ethBalance * 3500, // ETH price approx
      tokens: [
        {
          symbol: 'ETH',
          name: 'Ethereum',
          balance: ethBalance,
          valueUsd: ethBalance * 3500
        },
        ...validTokens
      ],
      change24h: 0
    });

  } catch (error) {
    console.error('Portfolio error:', error);
    return NextResponse.json({ error: 'Failed' }, { status: 500 });
  }
}

Create app/api/engagement/route.ts:

import { NextResponse } from 'next/server';
import { supabase } from '@/lib/supabase';

export async function POST(request: Request) {
  const { eventId, action, userId } = await request.json();

  try {
    // Record user action
    await supabase.from('event_user_engagement').insert({
      event_id: eventId,
      user_id: userId,
      action
    }).onConflict('event_id,user_id,action').ignore();

    // Increment count
    const column = action === 'view' ? 'views' : action === 'like' ? 'likes' : 'shares';
    
    const { data } = await supabase
      .from('event_engagement')
      .select('*')
      .eq('event_id', eventId)
      .single();

    if (data) {
      await supabase
        .from('event_engagement')
        .update({ [column]: data[column] + 1 })
        .eq('event_id', eventId);
    } else {
      await supabase.from('event_engagement').insert({
        event_id: eventId,
        [column]: 1
      });
    }

    // Get updated counts
    const { data: updated } = await supabase
      .from('event_engagement')
      .select('*')
      .eq('event_id', eventId)
      .single();

    return NextResponse.json(updated);
  } catch (error) {
    return NextResponse.json({ error: 'Failed' }, { status: 500 });
  }
}

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const eventId = searchParams.get('eventId');

  const { data } = await supabase
    .from('event_engagement')
    .select('*')
    .eq('event_id', eventId)
    .single();

  return NextResponse.json(data || { views: 0, likes: 0, shares: 0, comments: 0 });
}
