// Compare two snapshots to detect holder changes

export async function detectHolderChanges(
  tokenAddress: string,
  olderSnapshotId: string,
  newerSnapshotId: string
): Promise<{
  newEntries: any[];
  exits: any[];
  increases: any[];
  decreases: any[];
}> {
  const { data: older } = await supabaseAdmin
    .from('holder_snapshots')
    .select('*')
    .eq('id', olderSnapshotId)
    .single();

  const { data: newer } = await supabaseAdmin
    .from('holder_snapshots')
    .select('*')
    .eq('id', newerSnapshotId)
    .single();

  if (!older || !newer) throw new Error('Snapshots not found');

  const olderMap = new Map(older.holders.map((h: any) => [h.address, h]));
  const newerMap = new Map(newer.holders.map((h: any) => [h.address, h]));

  const newEntries: any[] = [];
  const exits: any[] = [];
  const increases: any[] = [];
  const decreases: any[] = [];

  // Find new entries
  for (const [address, holder] of newerMap) {
    if (!olderMap.has(address)) {
      newEntries.push(holder);
    }
  }

  // Find exits
  for (const [address, holder] of olderMap) {
    if (!newerMap.has(address)) {
      exits.push(holder);
    }
  }

  // Find increases/decreases
  for (const [address, newHolder] of newerMap) {
    const oldHolder = olderMap.get(address);
    if (oldHolder) {
      const oldPercentage = oldHolder.percentage;
      const newPercentage = newHolder.percentage;

      if (newPercentage > oldPercentage) {
        increases.push({
          ...newHolder,
          change: newPercentage - oldPercentage,
        });
      } else if (newPercentage < oldPercentage) {
        decreases.push({
          ...newHolder,
          change: oldPercentage - newPercentage,
        });
      }
    }
  }

  return { newEntries, exits, increases, decreases };
}

export async function getEntityHistory(entityId: string): Promise<EntityHistory> {
  const { data: trades } = await supabaseAdmin
    .from('entity_trades')
    .select('*')
    .eq('entity_id', entityId)
    .order('entry_date', { ascending: false });

  if (!trades || trades.length === 0) {
    return {
      entityId,
      entityName: 'Unknown',
      trades: [],
      performance: {
        totalTrades: 0,
        wins: 0,
        losses: 0,
        winRate: 0,
        avgHoldTime: 0,
        avgGain: 0,
        avgLoss: 0,
        bestTrade: {} as EntityTrade,
        worstTrade: {} as EntityTrade,
      },
    };
  }

  const entityName = trades[0].entity_name;
  const completedTrades = trades.filter((t: any) => t.status === 'EXITED');
  
  const wins = completedTrades.filter((t: any) => t.gain > 0);
  const losses = completedTrades.filter((t: any) => t.gain <= 0);

  const avgHoldTime = completedTrades.reduce((sum: number, t: any) => 
    sum + t.hold_time, 0
  ) / completedTrades.length;

  const avgGain = wins.length > 0
    ? wins.reduce((sum: number, t: any) => sum + t.gain, 0) / wins.length
    : 0;

  const avgLoss = losses.length > 0
    ? losses.reduce((sum: number, t: any) => sum + t.gain, 0) / losses.length
    : 0;

  const bestTrade = completedTrades.reduce((best: any, t: any) => 
    !best || t.gain > best.gain ? t : best
  , null);

  const worstTrade = completedTrades.reduce((worst: any, t: any) =>
    !worst || t.gain < worst.gain ? t : worst
  , null);

  return {
    entityId,
    entityName,
    trades: trades.map(mapTradeFromDB),
    performance: {
      totalTrades: completedTrades.length,
      wins: wins.length,
      losses: losses.length,
      winRate: (wins.length / completedTrades.length) * 100,
      avgHoldTime,
      avgGain,
      avgLoss,
      bestTrade: mapTradeFromDB(bestTrade),
      worstTrade: mapTradeFromDB(worstTrade),
    },
  };
}

    .select('*')
    .eq('status', 'PENDING')
    .lt('expires_at', new Date().toISOString());

  for (const order of orders) {
    try {
      // Get current price (would use price feed)

      // Check if target price reached
      let shouldExecute = false;
      if (order.side === 'BUY' && currentPrice <= order.target_price) {

function calculateNextBuyTime(frequency: 'HOURLY' | 'DAILY' | 'WEEKLY'): Date {
  const now = new Date();
  switch (frequency) {
    case 'HOURLY':
      now.setHours(now.getHours() + 1);
      break;
    case 'DAILY':
      now.setDate(now.getDate() + 1);
      break;
    case 'WEEKLY':
      now.setDate(now.getDate() + 7);
      break;
  }
  return now;
}

class BackgroundJobManager {
  private intervals: Map<string, NodeJS.Timeout> = new Map();

  // =====================================================================
  // Start all background jobs

  startAll() {
    console.log('🚀 Starting all background jobs...');

    // Money Radar monitoring (every 1 minute)

    }, 60 * 1000);

    // Limit order monitoring (every 30 seconds)
    this.startJob('limitOrders', async () => {
      await monitorLimitOrders();
    }, 30 * 1000);

    // Snapshot popular tokens (every 1 hour)
    this.startJob('snapshots', async () => {
      await this.snapshotPopularTokens();
    }, 60 * 60 * 1000);

    // Update entity cache (every 6 hours)
    this.startJob('entityCache', async () => {
      await this.updateEntityCache();
    }, 6 * 60 * 60 * 1000);

    // Clean expired data (every 24 hours)
    this.startJob('cleanup', async () => {
      await this.cleanupExpiredData();
    }, 24 * 60 * 60 * 1000);
  }

  // =====================================================================
  // Stop all jobs
  // =====================================================================
  stopAll() {
    console.log('🛑 Stopping all background jobs...');
    for (const [name, interval] of this.intervals) {
      clearInterval(interval);
      console.log(`Stopped job: ${name}`);
    }
    this.intervals.clear();
  }

  // =====================================================================
  // Start individual job
  // =====================================================================
  private startJob(name: string, handler: () => Promise<void>, intervalMs: number) {
    // Run immediately
    handler().catch(err => console.error(`Job ${name} failed:`, err));

    // Then run on interval
    const interval = setInterval(async () => {
      try {
        await handler();
      } catch (error) {
        console.error(`Job ${name} failed:`, error);
      }
    }, intervalMs);

    this.intervals.set(name, interval);
    console.log(`✅ Started job: ${name} (interval: ${intervalMs}ms)`);
  }

import { WebSocketServer, WebSocket } from 'ws';
import { priceFeedManager } from '../market/priceFeeds';

class RealtimeServer {
  private wss: WebSocketServer | null = null;
  private clients = new Map<string, WebSocket>();

  // =====================================================================
  // Start WebSocket server
  // =====================================================================
  start(port: number = 8080) {
    this.wss = new WebSocketServer({ port });

    this.wss.on('connection', (ws: WebSocket, req: any) => {
      const clientId = this.generateClientId();
      this.clients.set(clientId, ws);

      console.log(`✅ Client connected: ${clientId} (Total: ${this.clients.size})`);

      ws.on('message', (message: string) => {
        this.handleMessage(clientId, message);
      });

      ws.on('close', () => {
        this.clients.delete(clientId);
        console.log(`❌ Client disconnected: ${clientId}`);
      });

      // Send welcome message
      this.send(clientId, {
        type: 'CONNECTED',
        clientId,
        message: 'Welcome to STEINZ Labs realtime server',
      });
    });

    console.log(`🌐 WebSocket server started on port ${port}`);
  }

  // =====================================================================
  // Handle incoming messages
  // =====================================================================
  private handleMessage(clientId: string, message: string) {
    try {
      const data = JSON.parse(message);

      switch (data.type) {
        case 'SUBSCRIBE_PRICE':
          this.subscribeToPriceUpdates(clientId, data.tokenAddress);
          break;
        
        case 'UNSUBSCRIBE_PRICE':
          // Unsubscribe logic
          break;

        case 'SUBSCRIBE_ENTITY':
          // Subscribe to entity activity
          break;

        default:
          console.log('Unknown message type:', data.type);
      }
    } catch (error) {
      console.error('Failed to parse message:', error);
    }
  }

  // =====================================================================
  // Subscribe client to price updates
  // =====================================================================
  private subscribeToPriceUpdates(clientId: string, tokenAddress: string) {
    priceFeedManager.subscribe(tokenAddress, (update) => {
      this.send(clientId, {
        type: 'PRICE_UPDATE',
        tokenAddress,
        data: update,
      });
    });
  }

  // =====================================================================
  // Send message to specific client
  // =====================================================================
  private send(clientId: string, data: any) {
    const client = this.clients.get(clientId);
    if (client && client.readyState === WebSocket.OPEN) {
      client.send(JSON.stringify(data));
    }
  }

  // =====================================================================
  // Broadcast to all clients
  // =====================================================================
  broadcast(data: any) {
    const message = JSON.stringify(data);
    this.clients.forEach((client) => {
      if (client.readyState === WebSocket.OPEN) {
        client.send(message);
      }
    });
  }

  // =====================================================================
  // Generate unique client ID
  // =====================================================================
  private generateClientId(): string {
    return `client_${Date.now()}\_${Math.random().toString(36).substr(2, 9)}`;
  }
}

export const realtimeServer = new RealtimeServer();
