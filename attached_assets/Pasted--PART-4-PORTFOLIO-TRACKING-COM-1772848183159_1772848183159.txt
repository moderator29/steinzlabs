app/api/portfolio/route.ts:

import { NextRequest, NextResponse } from 'next/server';
import { getUserFromToken } from '@/lib/auth';
import { supabase } from '@/lib/supabase';
import { getTokenBalances } from '@/services/blockchain/portfolio';
import { getTokenPrice } from '@/services/analytics/coingecko';

export async function GET(request: NextRequest) {
  try {
    const user = await getUserFromToken(request);
    if (!user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    // Get user's wallet address
    const { data: userData } = await supabase
      .from('users')
      .select('wallet_address')
      .eq('id', user.id)
      .single();
    
    if (!userData?.wallet_address) {
      return NextResponse.json({ error: 'No wallet connected' }, { status: 400 });
    }
    
    // Fetch balances from blockchain for all chains
    const chains = ['ethereum', 'solana', 'base', 'arbitrum', 'polygon'];
    
    const allTokens = await Promise.all(
      chains.map(chain => getTokenBalances(userData.wallet_address, chain))
    );
    
    const tokens = allTokens.flat();
    
    // Calculate totals
    const totalValue = tokens.reduce((sum, token) => sum + (token.valueUsd || 0), 0);
    
    // Get 24h change
    const change24h = await calculate24hChange(tokens);
    
    // Get P&L data from database
    const { data: portfolioData } = await supabase
      .from('portfolios')
      .select('*')
      .eq('user_id', user.id);
    
    let totalInvested = 0;
    let realizedPnl = 0;
    let unrealizedPnl = 0;
    
    portfolioData?.forEach(position => {
      totalInvested += position.total_invested || 0;
      realizedPnl += position.realized_pnl || 0;
      unrealizedPnl += position.unrealized_pnl || 0;
    });
    
    const portfolioRoi = totalInvested > 0 
      ? ((totalValue - totalInvested) / totalInvested) * 100 
      : 0;
    
    return NextResponse.json({
      totalValue,
      change24h,
      totalInvested,
      realizedPnl,
      unrealizedPnl,
      portfolioRoi,
      tokens,
    });
  } catch (error: any) {
    console.error('Portfolio fetch error:', error);
    return NextResponse.json(
      { error: error.message },
      { status: 500 }
    );
  }
}

async function calculate24hChange(tokens: any[]) {
  let totalValueNow = 0;
  let totalValue24hAgo = 0;
  
  for (const token of tokens) {
    totalValueNow += token.valueUsd || 0;
    
    // Get price 24h ago
    const price24hAgo = token.price / (1 + (token.change24h / 100));
    const value24hAgo = token.balance * price24hAgo;
    totalValue24hAgo += value24hAgo;
  }
  
  const changeUsd = totalValueNow - totalValue24hAgo;
  const changePercent = totalValue24hAgo > 0 
    ? (changeUsd / totalValue24hAgo) * 100 
    : 0;
  
  return {
    usd: changeUsd,
    percent: changePercent,
  };
}

function calculateCopyScore(analysis: any): number {
  let score = 0;
  
  // Win rate (max 30 points)
  if (analysis.metrics.winRate >= 75) score += 30;
  else if (analysis.metrics.winRate >= 60) score += 20;
  else if (analysis.metrics.winRate >= 50) score += 10;
  
  // Profit factor (max 25 points)
  if (analysis.metrics.profitFactor >= 2.5) score += 25;
  else if (analysis.metrics.profitFactor >= 2.0) score += 20;
  else if (analysis.metrics.profitFactor >= 1.5) score += 15;
  
  // Risk management (max 20 points)
  if (analysis.riskProfile.usesStopLoss) score += 10;
  if (analysis.riskProfile.positionSizing === 'good') score += 10;
  
  // Activity (max 15 points)
  if (analysis.metrics.totalTrades >= 100) score += 15;
  else if (analysis.metrics.totalTrades >= 50) score += 10;
  else if (analysis.metrics.totalTrades >= 20) score += 5;
  
  // Consistency (max 10 points)
  if (analysis.patterns.consistency === 'high') score += 10;
  else if (analysis.patterns.consistency === 'medium') score += 5;
  
  return Math.min(100, score);
}

async function sendMessageToVTX({ message, context }: any) {
  const response = await fetch('/api/vtx', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ message, context }),
  });
  
  if (!response.ok) throw new Error('VTX request failed');
  return response.json();
}
