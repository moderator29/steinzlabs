async function checkContractDuplicate(contractAddress, chain) {
  // Normalize address (lowercase, remove spaces)
  const normalized = contractAddress.toLowerCase().trim();
  
  // Hash it
  const hash = sha256(normalized);
  
  // Check database
  const existing = await db.projects
    .where('contractHash', hash)
    .where('chain', chain)
    .first();
  
  if (existing) {
    return {
      isDuplicate: true,
      existingProject: existing,
      message: `This contract is already listed as "${existing.name}"`,
      link: `/discover/${existing.slug}`
    };
  }
  
  return { isDuplicate: false };
}

import { perceptualHash } from 'image-hash';

async function checkLogoDuplicate(logoFile) {
  // Generate perceptual hash (detects similar images even if resized/compressed)
  const hash = await perceptualHash(logoFile);
  
  // Check against all existing logos
  const allLogos = await db.projects.select('logoHash', 'name', 'slug');
  
  for (const existing of allLogos) {
    const similarity = hammingDistance(hash, existing.logoHash);
    
    // If <10% different, it's a match
    if (similarity < 0.1) {
      return {
        isDuplicate: true,
        existingProject: existing,
        similarity: (1 - similarity) * 100, // e.g., 95% similar
        message: `Logo is 95% similar to "${existing.name}"`
      };
    }
  }
  
  return { isDuplicate: false, hash };
}

import { levenshteinDistance } from 'string-similarity';

async function checkNameDuplicate(projectName) {
  const normalized = projectName.toLowerCase().trim();
  const allNames = await db.projects.select('name', 'slug');
  
  for (const existing of allNames) {
    const existingNormalized = existing.name.toLowerCase().trim();
    const similarity = levenshteinDistance(normalized, existingNormalized);
    
    // If >80% similar, flag it
    if (similarity > 0.8) {
      return {
        isDuplicate: true,
        existingProject: existing,
        similarity: similarity * 100,
        message: `Name is ${similarity * 100}% similar to "${existing.name}"`
      };
    }
  }
  
  return { isDuplicate: false };
}

// Examples:

async function checkTeamDuplicate(founderWallet, teamWallets) {
  // Check if founder wallet has other projects
  const founderProjects = await db.projects
    .where('founderWallet', founderWallet)
    .select('name', 'slug', 'status');
  
  if (founderProjects.length > 0) {
    return {
      hasOtherProjects: true,
      projects: founderProjects,
      message: `This wallet has ${founderProjects.length} other project(s)`
    };
  }
  
  // Check team member wallets
  for (const wallet of teamWallets) {
    const teamProjects = await db.projectTeamMembers
      .where('wallet', wallet)
      .join('projects')
      .select('projects.name', 'projects.slug');
    
    if (teamProjects.length > 0) {
      return {
        hasOtherProjects: true,
        projects: teamProjects,
        teamMember: wallet
      };
    }
  }
  
  return { hasOtherProjects: false };
}

async function checkSocialDuplicate(twitter, discord, website) {

  // Twitter
  if (twitter) {
    const existing = await db.projects.where('twitter', twitter).first();
    if (existing) {
      checks.push({
        type: 'twitter',
        existing: existing.name,
        message: `Twitter @${twitter} already linked to "${existing.name}"`
      });
    }
  }
  
  // Discord
  if (discord) {
    const existing = await db.projects.where('discord', discord).first();
    if (existing) {
      checks.push({
        type: 'discord',
        existing: existing.name,
        message: `Discord already linked to "${existing.name}"`
      });
    }
  }
  
  // Website
  if (website) {
    const existing = await db.projects.where('website', website).first();
    if (existing) {
      checks.push({
        type: 'website',
        existing: existing.name,
        message: `Website already linked to "${existing.name}"`
      });
    }
  }
  
  return checks.length > 0 ? { isDuplicate: true, matches: checks } : { isDuplicate: false };
}

async function runFullDuplicateCheck(submission) {
  const results = {
    contract: await checkContractDuplicate(submission.contractAddress, submission.chain),
    logo: await checkLogoDuplicate(submission.logoFile),
    name: await checkNameDuplicate(submission.name),
    team: await checkTeamDuplicate(submission.founderWallet, submission.teamWallets),
    socials: await checkSocialDuplicate(submission.twitter, submission.discord, submission.website)
  };
  
  // If ANY layer detects duplicate
  if (results.contract.isDuplicate) {
    return {
      status: "BLOCKED",
      reason: "CONTRACT_DUPLICATE",
      details: results.contract
    };
  }
  
  if (results.logo.isDuplicate && results.logo.similarity > 90) {
    return {
      status: "BLOCKED",
      reason: "LOGO_DUPLICATE",
      details: results.logo
    };
  }
  
  if (results.name.isDuplicate && results.name.similarity > 90) {
    return {
      status: "WARNING",
      reason: "NAME_SIMILAR",
      details: results.name,
      action: "MANUAL_REVIEW" // You decide
    };
  }
  
  if (results.team.hasOtherProjects) {
    return {
      status: "INFO",
      reason: "TEAM_HAS_HISTORY",
      details: results.team,
      action: "FLAG_FOR_ADMIN" // Shown in admin panel
    };
  }
  
  if (results.socials.isDuplicate) {
    return {
      status: "BLOCKED",
      reason: "SOCIAL_DUPLICATE",
      details: results.socials
    };
  }
  
  return {
    status: "CLEAN",
    reason: "NO_DUPLICATES_FOUND"
  };
}

ACTION 1: APPROVE

async function approveSubmission(submissionId, adminId, notes) {
  // 1. Update submission status
  await db.projectSubmissions.update(submissionId, {
    status: 'approved',
    reviewed_at: new Date(),
    reviewed_by: adminId,
    admin_notes: notes
  });
  
  // 2. Create live project
  const submission = await db.projectSubmissions.findOne(submissionId);
  const project = await db.projects.create({
    ...submission,
    verified: true,
    verified_at: new Date(),
    status: 'active'
  });
  
  // 3. Log admin action
  await db.adminActions.create({
    admin_id: adminId,
    action: 'approved',
    submission_id: submissionId,
    timestamp: new Date(),
    notes: notes
  });
  
  // 4. Email submitter
  await sendEmail({
    to: submission.founder_email,
    subject: "🎉 Your Project is Live on STEINZ!",
    body: `
      Congratulations! Your project "${submission.name}" has been approved.
      
      View your project: https://steinz.io/discover/${project.slug}
      
      Next steps:
      - Share your project link
      - Respond to builder applications
      - Post regular updates
      
      Welcome to STEINZ!
    `
  });
  
  // 5. Send notification to internal Slack/Discord
  await notifyTeam({
    message: `✅ New project approved: ${submission.name} by ${adminId}`
  });
  
  return { success: true, projectUrl: `/discover/${project.slug}` };
}


ACTION 2: REQUEST MORE INFO

async function requestMoreInfo(submissionId, adminId, message, requiredFields) {
  // 1. Update submission
  await db.projectSubmissions.update(submissionId, {
    status: 'needs_info',
    admin_notes: `Requested: ${requiredFields.join(', ')}`
  });
  
  // 2. Reset 48-hour clock
  await db.projectSubmissions.update(submissionId, {
    submitted_at: new Date(), // Clock restarts when they provide info
    review_deadline: new Date(Date.now() + 48 * 60 * 60 * 1000)
  });
  
  // 3. Log action
  await db.adminActions.create({
    admin_id: adminId,
    action: 'requested_info',
    submission_id: submissionId,
    timestamp: new Date(),
    notes: message
  });
  
  // 4. Email submitter
  const submission = await db.projectSubmissions.findOne(submissionId);
  await sendEmail({
    to: submission.founder_email,
    subject: "More Information Needed for Your STEINZ Submission",
    body: `
      Hi ${submission.founder_name},
      
      We need additional information for "${submission.name}":
      
      ${message}
      
      Required:
      ${requiredFields.map(f => `- ${f}`).join('\n')}
      
      Update your submission here:
      https://steinz.io/submit/${submission.private_link}
      
      Once updated, we'll review within 48 hours.
    `
  });
  
  return { success: true };
}

AI RISK SCORING - HOW IT WORKS:


async function calculateRiskScore(submission) {
  let riskScore = 0; // Start at 0 (safe)
  const factors = [];
  
  // Anonymous team (+25 risk)
  if (submission.is_anonymous) {
    riskScore += 25;
    factors.push("Anonymous team");
  }
  
  // No whitepaper (+15 risk)
  if (!submission.whitepaper_url) {
    riskScore += 15;
    factors.push("No whitepaper");
  }
  
  // Contract not verified (+20 risk)
  const isVerified = await checkContractVerified(submission.contract_address);
  if (!isVerified) {
    riskScore += 20;
    factors.push("Contract not verified");
  }
  
  // No audit (+15 risk)
  if (!submission.audit_report_url) {
    riskScore += 15;
    factors.push("No security audit");
  }
  
  // New social accounts (+10 risk)
  const twitterAge = await getTwitterAccountAge(submission.twitter);
  if (twitterAge < 90) { // Less than 3 months
    riskScore += 10;
    factors.push("New Twitter account");
  }
  
  // GoPlus security check
  const goPlusResult = await scanTokenGoPlus(submission.contract_address);
  if (goPlusResult.is_honeypot) {
    riskScore += 50;
    factors.push("⚠️ HONEYPOT DETECTED");
  }
  if (goPlusResult.buy_tax > 10 || goPlusResult.sell_tax > 10) {
    riskScore += 20;
    factors.push("High tax detected");
  }
  
  // Logo similarity to known scams
  const logoCheck = await checkLogoDuplicate(submission.logo_file);
  if (logoCheck.isDuplicate && logoCheck.similarity > 80) {
    riskScore += 30;
    factors.push(`Logo similar to ${logoCheck.existingProject.name}`);
  }
  
  // Low holder count (+5 risk)
  if (goPlusResult.holder_count < 50) {
    riskScore += 5;
    factors.push("Low holder count");
  }
  
  // Calculate final score
  return {
    score: Math.min(riskScore, 100), // Cap at 100
    level: riskScore < 30 ? 'LOW' : riskScore < 60 ? 'MEDIUM' : riskScore < 80 ? 'HIGH' : 'VERY HIGH',
    factors: factors,
    recommendation: riskScore < 40 ? 'APPROVE' : riskScore < 70 ? 'MANUAL_REVIEW' : 'REJECT'
  };
}
