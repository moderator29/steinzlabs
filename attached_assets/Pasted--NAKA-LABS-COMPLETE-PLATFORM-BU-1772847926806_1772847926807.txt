-- Enable UUID extension
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- Users table
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  firebase_uid TEXT UNIQUE NOT NULL,
  username TEXT UNIQUE NOT NULL,
  display_name TEXT,
  email TEXT UNIQUE NOT NULL,
  avatar_url TEXT,
  bio TEXT,
  website TEXT,
  twitter_handle TEXT,
  wallet_address TEXT,
  naka_balance DECIMAL DEFAULT 0,
  tier INTEGER DEFAULT 0 CHECK (tier >= 0 AND tier <= 3),
  verified BOOLEAN DEFAULT FALSE,
  is_admin BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_users_wallet ON users(wallet_address);
CREATE INDEX idx_users_username ON users(username);
CREATE INDEX idx_users_tier ON users(tier);
CREATE INDEX idx_users_firebase ON users(firebase_uid);

-- User preferences
CREATE TABLE user_preferences (
  user_id UUID PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
  language TEXT DEFAULT 'en',
  theme TEXT DEFAULT 'dark',
  notifications_enabled BOOLEAN DEFAULT TRUE,
  email_alerts BOOLEAN DEFAULT TRUE,
  push_alerts BOOLEAN DEFAULT TRUE,
  telegram_alerts BOOLEAN DEFAULT FALSE,
  trading_confirmations BOOLEAN DEFAULT TRUE,
  auto_refresh BOOLEAN DEFAULT TRUE,
  preferences JSONB DEFAULT '{}'::jsonb,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Portfolios
CREATE TABLE portfolios (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  chain TEXT NOT NULL,
  token_address TEXT NOT NULL,
  token_symbol TEXT NOT NULL,
  token_name TEXT,
  balance DECIMAL NOT NULL,
  value_usd DECIMAL,
  avg_buy_price DECIMAL,
  total_invested DECIMAL,
  realized_pnl DECIMAL DEFAULT 0,
  unrealized_pnl DECIMAL DEFAULT 0,
  first_acquired TIMESTAMPTZ,
  last_updated TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, chain, token_address)
);

CREATE INDEX idx_portfolios_user ON portfolios(user_id);
CREATE INDEX idx_portfolios_chain ON portfolios(chain);
CREATE INDEX idx_portfolios_updated ON portfolios(last_updated DESC);

-- Signals (Context Feed data)
CREATE TABLE signals (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  signal_type TEXT NOT NULL,
  chain TEXT NOT NULL,
  token_address TEXT NOT NULL,
  token_symbol TEXT NOT NULL,
  token_name TEXT,
  wallet_address TEXT NOT NULL,
  wallet_label TEXT,
  wallet_type TEXT,
  amount DECIMAL NOT NULL,
  amount_usd DECIMAL NOT NULL,
  tx_hash TEXT NOT NULL UNIQUE,
  block_number BIGINT,
  trust_score INTEGER CHECK (trust_score >= 0 AND trust_score <= 100),
  sentiment TEXT CHECK (sentiment IN ('bullish', 'bearish', 'neutral')),
  title TEXT NOT NULL,
  summary TEXT NOT NULL,
  detailed_context TEXT,
  why_it_matters TEXT,
  historical_context TEXT,
  proof_data JSONB,
  metadata JSONB DEFAULT '{}'::jsonb,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_signals_chain ON signals(chain);
CREATE INDEX idx_signals_token ON signals(token_address);
CREATE INDEX idx_signals_wallet ON signals(wallet_address);
CREATE INDEX idx_signals_type ON signals(signal_type);
CREATE INDEX idx_signals_created ON signals(created_at DESC);
CREATE INDEX idx_signals_trust ON signals(trust_score DESC);
CREATE INDEX idx_signals_composite ON signals(chain, signal_type, created_at DESC);

-- Alerts
CREATE TABLE alerts (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  alert_type TEXT NOT NULL,
  name TEXT NOT NULL,
  description TEXT,
  conditions JSONB NOT NULL,
  enabled BOOLEAN DEFAULT TRUE,
  trigger_count INTEGER DEFAULT 0,
  last_triggered TIMESTAMPTZ,
  notification_channels TEXT[] DEFAULT ARRAY['platform'],
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_alerts_user ON alerts(user_id);
CREATE INDEX idx_alerts_enabled ON alerts(enabled) WHERE enabled = TRUE;

-- Alert triggers history
CREATE TABLE alert_triggers (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  alert_id UUID REFERENCES alerts(id) ON DELETE CASCADE,
  signal_id UUID REFERENCES signals(id),
  trigger_data JSONB,
  triggered_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_alert_triggers_alert ON alert_triggers(alert_id);
CREATE INDEX idx_alert_triggers_time ON alert_triggers(triggered_at DESC);

-- Watchlists
CREATE TABLE watchlists (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  description TEXT,
  is_public BOOLEAN DEFAULT FALSE,
  is_collaborative BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE watchlist_items (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  watchlist_id UUID REFERENCES watchlists(id) ON DELETE CASCADE,
  chain TEXT NOT NULL,
  token_address TEXT NOT NULL,
  token_symbol TEXT NOT NULL,
  token_name TEXT,
  notes TEXT,
  added_by UUID REFERENCES users(id),
  added_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_watchlists_user ON watchlists(user_id);
CREATE INDEX idx_watchlist_items_list ON watchlist_items(watchlist_id);

-- Social: Follows
CREATE TABLE follows (
  follower_id UUID REFERENCES users(id) ON DELETE CASCADE,
  following_id UUID REFERENCES users(id) ON DELETE CASCADE,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  PRIMARY KEY (follower_id, following_id),
  CHECK (follower_id != following_id)
);

CREATE INDEX idx_follows_follower ON follows(follower_id);
CREATE INDEX idx_follows_following ON follows(following_id);

-- Social: DM Conversations
CREATE TABLE dm_conversations (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  participant_1 UUID REFERENCES users(id) ON DELETE CASCADE,
  participant_2 UUID REFERENCES users(id) ON DELETE CASCADE,
  last_message_at TIMESTAMPTZ DEFAULT NOW(),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  CONSTRAINT unique_participants UNIQUE(participant_1, participant_2),
  CHECK (participant_1 < participant_2)
);

CREATE TABLE dm_messages (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  conversation_id UUID REFERENCES dm_conversations(id) ON DELETE CASCADE,
  sender_id UUID REFERENCES users(id) ON DELETE CASCADE,
  content TEXT NOT NULL,
  read BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_dm_conversations_p1 ON dm_conversations(participant_1);
CREATE INDEX idx_dm_conversations_p2 ON dm_conversations(participant_2);
CREATE INDEX idx_dm_messages_conversation ON dm_messages(conversation_id, created_at DESC);

-- Trades
CREATE TABLE trades (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  chain TEXT NOT NULL,
  tx_hash TEXT NOT NULL,
  trade_type TEXT CHECK (trade_type IN ('buy', 'sell')),
  token_in_address TEXT NOT NULL,
  token_in_symbol TEXT NOT NULL,
  token_in_amount DECIMAL NOT NULL,
  token_out_address TEXT NOT NULL,
  token_out_symbol TEXT NOT NULL,
  token_out_amount DECIMAL NOT NULL,
  price_usd DECIMAL,
  gas_fee_usd DECIMAL,
  platform_fee_usd DECIMAL,
  total_cost_usd DECIMAL,
  dex_aggregator TEXT,
  status TEXT DEFAULT 'pending' CHECK (status IN ('pending', 'confirmed', 'failed')),
  executed_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_trades_user ON trades(user_id);
CREATE INDEX idx_trades_executed ON trades(executed_at DESC);
CREATE INDEX idx_trades_status ON trades(status);

-- Achievements
CREATE TABLE achievements (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  code TEXT UNIQUE NOT NULL,
  name TEXT NOT NULL,
  description TEXT NOT NULL,
  icon TEXT,
  rarity TEXT CHECK (rarity IN ('common', 'rare', 'epic', 'legendary')),
  points INTEGER DEFAULT 0
);

CREATE TABLE user_achievements (
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  achievement_id UUID REFERENCES achievements(id) ON DELETE CASCADE,
  earned_at TIMESTAMPTZ DEFAULT NOW(),
  PRIMARY KEY (user_id, achievement_id)
);

-- Verification requests
CREATE TABLE verification_requests (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  full_name TEXT NOT NULL,
  reason TEXT NOT NULL,
  social_links JSONB,
  status TEXT DEFAULT 'pending' CHECK (status IN ('pending', 'approved', 'rejected')),
  reviewed_by UUID REFERENCES users(id),
  reviewed_at TIMESTAMPTZ,
  rejection_reason TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_verification_status ON verification_requests(status);
CREATE INDEX idx_verification_created ON verification_requests(created_at DESC);

-- Analytics & Activity Tracking
CREATE TABLE user_activity (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  activity_type TEXT NOT NULL,
  metadata JSONB DEFAULT '{}'::jsonb,
  ip_address TEXT,

// Real-time signal detection system
export class SignalPoller {
  private alchemyWs: WebSocket;
  private heliusWs: WebSocket;
  
  async start() {
    // Monitor Ethereum whale transfers
    this.monitorEthereumWhales();
    
    // Monitor Solana whale transfers
    this.monitorSolanaWhales();
    
    // Monitor DEX swaps for large trades
    this.monitorDEXSwaps();
    
    // Poll Dune for historical whale wallets
    this.pollDuneWhaleWallets();
  }
  
  private async monitorEthereumWhales() {
    // Use Alchemy WebSocket for real-time transfers
    const ws = new WebSocket(
      `wss://eth-mainnet.g.alchemy.com/v2/${ALCHEMY_API_KEY_ETH}`
    );
    
    ws.on('open', () => {
      // Subscribe to large ERC20 transfers
      ws.send(JSON.stringify({
        jsonrpc: '2.0',
        id: 1,
        method: 'eth_subscribe',
        params: ['logs', {
          topics: [
            '0xddf252ad1be2c89b69c2b068fc378daa952ba7f163c4a11628f55a4df523b3ef' // Transfer event
          ]
        }]
      }));
    });
    
    ws.on('message', async (data) => {
      const event = JSON.parse(data.toString());
      
      if (event.params?.result) {
        await this.processTransferEvent(event.params.result);
      }
    });
  }
  
  private async processTransferEvent(log: any) {
    // Decode transfer amount
    const amount = parseInt(log.data, 16);
    
    // Get token info
    const tokenInfo = await getTokenInfo(log.address);
    
    // Calculate USD value
    const usdValue = amount * tokenInfo.price / Math.pow(10, tokenInfo.decimals);
    
    // Only process large transfers (>$100k)
    if (usdValue < 100000) return;
    
    // Get wallet info
    const fromAddress = '0x' + log.topics[1].slice(26);
    const toAddress = '0x' + log.topics[2].slice(26);
    
    const walletInfo = await analyzeWallet(fromAddress, toAddress);
    
    // Determine signal type
    let signalType: string;
    if (walletInfo.isWhale) {
      signalType = walletInfo.isBuying ? 'whale_buy' : 'whale_sell';
    } else if (walletInfo.isSmartMoney) {
      signalType = walletInfo.isBuying ? 'smart_money_buy' : 'smart_money_sell';
    } else if (walletInfo.isDEX) {
      return; // Skip DEX internal transfers
    } else {
      signalType = 'large_transfer';
    }
    
    // Calculate trust score
    const trustScore = await calculateTrustScore({
      walletHistory: walletInfo.history,
      transactionSize: usdValue,
      tokenQuality: tokenInfo.quality,
      marketConditions: await getMarketConditions(),
    });
    
    // Generate AI context using Claude
    const context = await generateAIContext({
      signalType,
      token: tokenInfo,
      wallet: walletInfo,
      amount: usdValue,
      trustScore,
    });
    
    // Save signal to database
    await saveSignal({
      signalType,
      chain: 'ethereum',
      tokenAddress: log.address,
      tokenSymbol: tokenInfo.symbol,
      tokenName: tokenInfo.name,
      walletAddress: walletInfo.isBuying ? toAddress : fromAddress,
      walletLabel: walletInfo.label,
      walletType: walletInfo.type,
      amount: amount / Math.pow(10, tokenInfo.decimals),
      amountUsd: usdValue,
      txHash: log.transactionHash,
      blockNumber: parseInt(log.blockNumber, 16),
      trustScore,
      sentiment: determineSentiment(signalType, walletInfo, tokenInfo),
      title: context.title,
      summary: context.summary,
      detailedContext: context.detailed,
      whyItMatters: context.whyItMatters,
      historicalContext: context.historical,
      proofData: {
        transactions: [{ hash: log.transactionHash, ... }],
        wallets: [walletInfo],
        sources: ['Alchemy', 'Etherscan', 'Dune Analytics'],
      },
    });
    
    // Trigger real-time update to frontend via WebSocket
    broadcastSignal(signal);
    
    // Check if any user alerts match this signal
    await checkAndTriggerAlerts(signal);
  }
}
