'use client';

import { useEffect, useRef, useState } from 'react';
import { createChart } from 'lightweight-charts';

export function TradingChart({ token, onPriceClick, dragPrice }) {
  const chartContainerRef = useRef<HTMLDivElement>(null);
  const chartRef = useRef<any>(null);
  const [isDragging, setIsDragging] = useState(false);
  
  useEffect(() => {
    if (!chartContainerRef.current) return;
    
    const chart = createChart(chartContainerRef.current, {
      layout: {
        background: { color: '#0A0E1A' },
        textColor: '#A0AEC0',
      },
      width: chartContainerRef.current.clientWidth,
      height: chartContainerRef.current.clientHeight,
      rightPriceScale: {
        borderVisible: false,
      },
      timeScale: {
        borderVisible: false,
      },
      crosshair: {
        mode: 1, // Normal crosshair
      },
    });
    
    const candlestickSeries = chart.addCandlestickSeries({
      upColor: '#00E676',
      downColor: '#FF4B4B',
      borderVisible: false,
      wickUpColor: '#00E676',
      wickDownColor: '#FF4B4B',
    });
    
    // Load chart data
    fetchChartData(token).then(data => {
      candlestickSeries.setData(data.candlesticks);
    });
    
    // CLICK TO SET PRICE LINE
    chart.subscribeClick((param) => {
      if (param.point && param.seriesPrices) {
        const price = param.seriesPrices.get(candlestickSeries);
        if (price) {
          onPriceClick(price);
        }
      }
    });
    
    chartRef.current = chart;
    
    return () => chart.remove();
  }, [token]);
  
  // Draw price line if dragged
  useEffect(() => {
    if (!chartRef.current || !dragPrice) return;
    
    const priceLine = chartRef.current.addLineSeries({
      color: '#00E5FF',
      lineWidth: 2,
      lineStyle: 2, // Dashed
      priceLineVisible: false,
    });
    
    priceLine.setData([
      { time: Date.now() / 1000, value: dragPrice },
    ]);
    
    return () => priceLine.remove();
  }, [dragPrice]);
  
  return (
    <div ref={chartContainerRef} className="w-full h-full" />
  );
}

export async function executeOrder(order: Order) {
  const startTime = Date.now();
  
  try {
    // 1. Build transaction
    const tx = await buildSwapTransaction(order);
    
    // 2. MEV PROTECTION - Use Flashbots/private RPC
    const protectedRPC = process.env.FLASHBOTS_RPC || HELIUS_RPC;
    
    // 3. Simulate first
    const simulation = await simulateTransaction(tx, protectedRPC);
    
    if (!simulation.success) {
      throw new Error(`Simulation failed: ${simulation.error}`);
    }
    
    // 4. Sign and send (user wallet signs)
    const signature = await userWallet.signTransaction(tx);
    
    // 5. Send via protected RPC
    const txHash = await sendTransaction(signature, protectedRPC);
    
    // 6. Log execution time
    const executionTime = Date.now() - startTime;
    console.log(`Order executed in ${executionTime}ms`);
    
    // 7. Update database immediately (don't wait for confirmation)
    await logTrade({
      ...order,
      txHash,
      status: 'pending',
      executionTimeMs: executionTime,
    });
    
    // 8. Return optimistic result
    return {
      success: true,
      txHash,
      executionTime,
      expectedOutput: simulation.expectedOutput,
    };
    
  } catch (error) {
    console.error('Execution failed:', error);
    throw error;
  }
}

5. REAL-TIME P&L TRACKING:

services/analytics/pnlTracker.ts:

export function calculateRealTimePnL(portfolio: Portfolio) {
  let unrealizedPnL = 0;
  let realizedPnL = 0;
  
  // For each position
  portfolio.positions.forEach(position => {
    const currentValue = position.amount * position.currentPrice;
    const costBasis = position.amount * position.averageEntryPrice;
    
    // Unrealized P&L
    unrealizedPnL += (currentValue - costBasis);
  });
  
  // For closed trades
  portfolio.closedTrades.forEach(trade => {
    realizedPnL += trade.profitLoss;
  });
  
  return {
    unrealizedPnL,
    realizedPnL,
    totalPnL: unrealizedPnL + realizedPnL,
    roi: ((unrealizedPnL + realizedPnL) / portfolio.totalInvested) * 100,
  };
}
