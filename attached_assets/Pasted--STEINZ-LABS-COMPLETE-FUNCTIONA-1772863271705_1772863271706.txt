'use client';

import { useState } from 'react';
import { useQuery } from '@tanstack/react-query';
import { Card } from '@/components/ui/Card';
import { Select } from '@/components/ui/Select';
import { Input } from '@/components/ui/Input';

export function HistoryTab() {
  const [filter, setFilter] = useState('all'); // all, sent, received, swap
  const [search, setSearch] = useState('');
  
  const { data: transactions } = useQuery({
    queryKey: ['transactions', filter],
    queryFn: () => fetchTransactions(filter),
  });
  
  const filteredTxs = transactions?.filter(tx =>
    tx.tokenSymbol.toLowerCase().includes(search.toLowerCase()) ||
    tx.hash.toLowerCase().includes(search.toLowerCase())
  );
  
  return (
    <div className="space-y-4">
      {/* Filters */}
      <div className="flex gap-3">
        <Select
          value={filter}
          onChange={setFilter}
          options={[
            { value: 'all', label: 'All Transactions' },
            { value: 'sent', label: 'Sent' },
            { value: 'received', label: 'Received' },
            { value: 'swap', label: 'Swaps' },
          ]}
        />
        <Input
          placeholder="Search by token or hash..."
          value={search}
          onChange={(e) => setSearch(e.target.value)}
        />
      </div>
      
      {/* Transaction List */}
      <div className="space-y-2">
        {filteredTxs?.map((tx) => (
          <TransactionCard key={tx.hash} transaction={tx} />
        ))}
      </div>
    </div>
  );
}

function TransactionCard({ transaction }) {
  const getIcon = (type) => {
    switch (type) {
      case 'sent': return '↗️';
      case 'received': return '↙️';
      case 'swap': return '🔄';
      default: return '📝';
    }
  };
  
  return (
    <Card className="p-4 hover:border-accent-primary/50 transition-colors">
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-3">
          <div className="text-2xl">{getIcon(transaction.type)}</div>
          <div>
            <p className="font-medium text-text-primary">
              {transaction.type === 'swap' 
                ? `Swap ${transaction.tokenIn} → ${transaction.tokenOut}`
                : `${transaction.type} ${transaction.tokenSymbol}`
              }
            </p>
            <p className="text-sm text-text-tertiary">
              {new Date(transaction.timestamp).toLocaleString()}
            </p>
          </div>
        </div>
        
        <div className="text-right">
          <p className={`font-medium ${
            transaction.type === 'received' ? 'text-success' : 'text-text-primary'
          }`}>
            {transaction.type === 'received' ? '+' : '-'}
            {transaction.amount} {transaction.tokenSymbol}
          </p>
          <p className="text-sm text-text-tertiary">
            ${transaction.valueUSD.toFixed(2)}
          </p>
        </div>
      </div>
      
      <a
        href={`https://etherscan.io/tx/${transaction.hash}`}
        target="_blank"
        className="text-xs text-accent-primary hover:underline mt-2 block"
      >
        View on Explorer →
      </a>
    </Card>
  );
}

async function createNonCustodialWallet() {
  // CRITICAL: Generate wallet CLIENT-SIDE ONLY
  // NEVER send private key to server
  
  const wallet = ethers.Wallet.createRandom();
  
  // Show recovery phrase to user (they MUST save it)
  const mnemonic = wallet.mnemonic.phrase;
  
  // Encrypt private key with user password before storing locally
  const encrypted = await encryptPrivateKey(wallet.privateKey, userPassword);
  
  // Store ONLY encrypted version in browser localStorage
  localStorage.setItem(`wallet_${wallet.address}`, encrypted);
  
  // Send ONLY public address to backend
  await fetch('/api/wallets', {
    method: 'POST',
    body: JSON.stringify({ address: wallet.address }),
  });
  
  return { address: wallet.address, mnemonic };
}

'use client';

import { useState } from 'react';
import { useMutation } from '@tanstack/react-query';

export default function DNAAnalyzerPage() {
  const [address, setAddress] = useState('');
  const [addressType, setAddressType] = useState<'wallet' | 'contract' | null>(null);
  
  const analyzeMutation = useMutation({
    mutationFn: analyzeAddress,
    onError: (error: any) => {
      if (error.message.includes('contract')) {
        setAddressType('contract');
      }
    },
    onSuccess: (data) => {
      setAddressType('wallet');
    },
  });
  
  const handleAnalyze = async () => {
    setAddressType(null);
    
    // First detect if it's a contract
    const isContract = await checkIfContract(address);
    
    if (isContract) {
      setAddressType('contract');
      return;
    }
    
    // If it's a wallet, proceed with analysis
    analyzeMutation.mutate(address);
  };
  
  return (
    <div className="max-w-5xl mx-auto px-4 py-6">
      <h1 className="text-3xl font-bold mb-2">DNA Analyzer</h1>
      <p className="text-text-secondary mb-6">
        Analyze wallet trading behavior and patterns
      </p>
      
      {/* Input */}
      <Card className="p-6 mb-6">
        <Input
          placeholder="Enter WALLET address (not contract address)"
          value={address}
          onChange={(e) => setAddress(e.target.value)}
          className="mb-4"
        />
        <Button onClick={handleAnalyze} disabled={!address}>
          Analyze Wallet
        </Button>
      </Card>
      
      {/* Contract Address Error */}
      {addressType === 'contract' && (
        <Card className="p-6 border-error/50 bg-error/5">
          <div className="flex items-start gap-3">
            <AlertCircleIcon className="w-6 h-6 text-error flex-shrink-0" />

                ❌ This is a Smart Contract Address
              </h3>
              <p className="text-text-secondary mb-4">
                DNA Analyzer only works with WALLET addresses, not smart contracts.
              </p>
              <p className="text-sm text-text-secondary mb-2">
                <strong>What's the difference?</strong>
              </p>
              <ul className="text-sm text-text-secondary space-y-1 list-disc list-inside">
                <li><strong>Wallet Address:</strong> Controlled by a person with private keys (can trade, hold tokens)</li>
                <li><strong>Contract Address:</strong> Smart contract code deployed on blockchain (tokens, protocols, etc.)</li>

// Check if address is a contract
async function checkIfContract(address: string): Promise<boolean> {
  const provider = new ethers.JsonRpcProvider(ALCHEMY_RPC_URL);
  const code = await provider.getCode(address);
  
  // If code exists, it's a contract
  // Wallets have code = "0x"
  return code !== '0x';
}

'use client';

import { useEffect, useRef } from 'react';
import { createChart } from 'lightweight-charts';

export function TradingChart({ tokenAddress, chain }) {
  const chartContainerRef = useRef<HTMLDivElement>(null);
  const chartRef = useRef<any>(null);
  
  useEffect(() => {
    if (!chartContainerRef.current) return;
    
    // Create chart
    const chart = createChart(chartContainerRef.current, {
      layout: {
        background: { color: '#0A0E1A' },
        textColor: '#A0AEC0',
      },
      grid: {
        vertLines: { color: '#1E2433' },
        horzLines: { color: '#1E2433' },
      },
      width: chartContainerRef.current.clientWidth,
      height: 500,
    });
    
    // Add candlestick series
    const candlestickSeries = chart.addCandlestickSeries({
      upColor: '#00E676',
      downColor: '#FF4B4B',
      borderVisible: false,
      wickUpColor: '#00E676',
      wickDownColor: '#FF4B4B',
    });
    
    // Add volume series
    const volumeSeries = chart.addHistogramSeries({
      color: '#00E5FF',
      priceFormat: {
        type: 'volume',
      },
      priceScaleId: '',
    });
    
    volumeSeries.priceScale().applyOptions({
      scaleMargins: {
        top: 0.8,
        bottom: 0,
      },
    });
    
    // Fetch and set data
    fetchChartData(tokenAddress, chain).then(data => {
      candlestickSeries.setData(data.candlesticks);
      volumeSeries.setData(data.volumes);
    });
    
    chartRef.current = chart;
    
    // Cleanup
    return () => chart.remove();
  }, [tokenAddress, chain]);
  
  return (
    <div className="space-y-4">
      {/* Chart Controls */}
      <div className="flex items-center justify-between">
        <div className="flex gap-2">
          {['1H', '4H', '1D', '1W', '1M'].map((interval) => (
            <button
              key={interval}
              className="px-3 py-1 rounded bg-background-elevated hover:bg-background-hover text-sm"
            >
              {interval}
            </button>
          ))}
        </div>
        
        <div className="flex gap-2">
          <button className="px-3 py-1 rounded bg-background-elevated hover:bg-background-hover text-sm">
            📊 Indicators
          </button>
          <button className="px-3 py-1 rounded bg-background-elevated hover:bg-background-hover text-sm">
            🖊️ Draw
          </button>
        </div>
      </div>
      
      {/* Chart */}
      <div ref={chartContainerRef} className="rounded border border-border" />
    </div>
  );
}

export async function autoDetectAndAssignTier(walletAddress: string, userId: string) {
  // Check NAKA balance
  const nakaBalance = await getNakaBalance(walletAddress);
  
  // Calculate tier
  let tier = 0;
  if (nakaBalance >= 2000000) tier = 3; // Gold
  else if (nakaBalance >= 1000000) tier = 2; // Silver
  else if (nakaBalance >= 500000) tier = 1; // Bronze
  
  // Update user in database
  await supabase
    .from('users')
    .update({
      wallet_address: walletAddress,
      naka_balance: nakaBalance,
      tier: tier,
    })
    .eq('id', userId);
  
  // Unlock features based on tier
  return {
    tier,
    tierName: getTierName(tier),
    unlockedFeatures: getUnlockedFeatures(tier),
  };
}

function getUnlockedFeatures(tier: number) {
  const features = {
    0: ['dashboard', 'market', 'portfolio', 'scanner'],
    1: ['dashboard', 'market', 'portfolio', 'scanner', 'context_feed', 'alerts', 'watchlists'],
    2: ['dashboard', 'market', 'portfolio', 'scanner', 'context_feed', 'alerts', 'watchlists', 'dna_analyzer', 'advanced_charts'],
    3: ['all_features', 'priority_support', 'api_access', 'copy_trading'],
  };
  
  return features[tier];
}
