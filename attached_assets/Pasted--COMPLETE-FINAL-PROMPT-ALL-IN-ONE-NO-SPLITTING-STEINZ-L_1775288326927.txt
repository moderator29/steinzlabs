import axios from 'axios';

export interface TokenMetadata {
  address: string;
  symbol: string;
  name: string;
  logo: string | null;
  decimals: number;
  price: number;
  priceChange24h: number;
  volume24h: number;
  marketCap: number;
  chain: string;
}

const metadataCache = new Map<string, { data: TokenMetadata; expires: number }>();

export async function getTokenMetadata(address: string, chain: string = 'solana'): Promise<TokenMetadata> {
  const cacheKey = `${chain}:${address}`;
  const cached = metadataCache.get(cacheKey);
  if (cached && cached.expires > Date.now()) return cached.data;

  try {
    const cgData = await fetchFromCoinGecko(address, chain);
    if (cgData) return cacheAndReturn(cacheKey, cgData);

    const dexData = await fetchFromDEXScreener(address, chain);
    if (dexData) return cacheAndReturn(cacheKey, dexData);

    if (chain === 'solana') {
      const heliusData = await fetchFromHelius(address);
      if (heliusData) return cacheAndReturn(cacheKey, heliusData);
    }

    return cacheAndReturn(cacheKey, createFallback(address, chain));
  } catch (error) {
    console.error('Token metadata fetch failed:', error);
    return createFallback(address, chain);
  }
}

async function fetchFromCoinGecko(address: string, chain: string): Promise<TokenMetadata | null> {
  try {
    const platformMap: Record<string, string> = { solana: 'solana', ethereum: 'ethereum', polygon: 'polygon-pos', arbitrum: 'arbitrum-one', optimism: 'optimistic-ethereum', base: 'base', bsc: 'binance-smart-chain' };
    const platform = platformMap[chain];
    if (!platform) return null;

    const res = await axios.get(`https://api.coingecko.com/api/v3/coins/${platform}/contract/${address}`, { timeout: 3000 });
    const d = res.data;
    return { address, symbol: d.symbol?.toUpperCase() || 'UNKNOWN', name: d.name || 'Unknown', logo: d.image?.large || d.image?.small, decimals: d.detail_platforms?.[platform]?.decimal_place || 18, price: d.market_data?.current_price?.usd || 0, priceChange24h: d.market_data?.price_change_percentage_24h || 0, volume24h: d.market_data?.total_volume?.usd || 0, marketCap: d.market_data?.market_cap?.usd || 0, chain };
  } catch { return null; }
}

async function fetchFromDEXScreener(address: string, chain: string): Promise<TokenMetadata | null> {
  try {
    const res = await axios.get(`https://api.dexscreener.com/latest/dex/tokens/${address}`, { timeout: 3000 });
    const pairs = res.data.pairs || [];
    if (!pairs.length) return null;
    const best = pairs.reduce((a: any, b: any) => parseFloat(a.liquidity?.usd || '0') > parseFloat(b.liquidity?.usd || '0') ? a : b);
    const token = best.baseToken;
    return { address, symbol: token.symbol || 'UNKNOWN', name: token.name || 'Unknown', logo: best.info?.imageUrl, decimals: 9, price: parseFloat(best.priceUsd || '0'), priceChange24h: parseFloat(best.priceChange?.h24 || '0'), volume24h: parseFloat(best.volume?.h24 || '0'), marketCap: parseFloat(best.marketCap || '0'), chain: best.chainId || chain };
  } catch { return null; }
}

async function fetchFromHelius(address: string): Promise<TokenMetadata | null> {
  try {
    const res = await axios.get(`https://api.helius.xyz/v0/token-metadata`, { params: { 'api-key': process.env.HELIUS_API_KEY, mintAddress: address }, timeout: 3000 });
    const d = res.data;
    return { address, symbol: d.symbol || 'UNKNOWN', name: d.name || 'Unknown', logo: d.logoURI, decimals: d.decimals || 9, price: 0, priceChange24h: 0, volume24h: 0, marketCap: 0, chain: 'solana' };
  } catch { return null; }
}

function cacheAndReturn(key: string, data: TokenMetadata): TokenMetadata {
  metadataCache.set(key, { data, expires: Date.now() + 3600000 });
  return data;
}

function createFallback(address: string, chain: string): TokenMetadata {
  return { address, symbol: address.slice(0, 6).toUpperCase(), name: `Token ${address.slice(0, 8)}`, logo: null, decimals: chain === 'solana' ? 9 : 18, price: 0, priceChange24h: 0, volume24h: 0, marketCap: 0, chain };
}

export async function batchGetTokenMetadata(tokens: Array<{ address: string; chain: string }>): Promise<Map<string, TokenMetadata>> {
  const results = new Map<string, TokenMetadata>();
  const batchSize = 10;
  for (let i = 0; i < tokens.length; i += batchSize) {
    const batch = tokens.slice(i, i + batchSize);
    await Promise.all(batch.map(async (t) => {
      const meta = await getTokenMetadata(t.address, t.chain);
      results.set(t.address, meta);
    }));
    if (i + batchSize < tokens.length) await new Promise(r => setTimeout(r, 200));
  }
  return results;
}

export interface WalletSession {
  userId: string;
  walletAddress: string;
  chain: string;
  authenticated: boolean;
  biometricEnabled: boolean;
  lastAuth: Date;
}

export async function checkWalletSession(): Promise<WalletSession | null> {
  try {
    const sessionData = localStorage.getItem('steinz_wallet_session');
    if (!sessionData) return null;
    const session = JSON.parse(sessionData);
    const hoursSinceAuth = (Date.now() - new Date(session.lastAuth).getTime()) / 3600000;
    if (hoursSinceAuth > 24) { localStorage.removeItem('steinz_wallet_session'); return null; }
    return session;
  } catch { return null; }
}

export async function createWalletSession(userId: string, walletAddress: string, chain: string): Promise<WalletSession> {
  const session: WalletSession = { userId, walletAddress, chain, authenticated: true, biometricEnabled: await checkBiometricSupport(), lastAuth: new Date() };
  localStorage.setItem('steinz_wallet_session', JSON.stringify(session));
  return session;
}

export async function refreshWalletSession(): Promise<WalletSession | null> {
  const session = await checkWalletSession();
  if (!session) return null;
  if (session.biometricEnabled) {
    const authenticated = await requestBiometricAuth();
    if (!authenticated) return null;
  }
  session.lastAuth = new Date();
  localStorage.setItem('steinz_wallet_session', JSON.stringify(session));
  return session;
}

export function destroyWalletSession(): void {
  localStorage.removeItem('steinz_wallet_session');
}

async function checkBiometricSupport(): Promise<boolean> {
  if (!window.PublicKeyCredential) return false;
  try { return await PublicKeyCredential.isUserVerifyingPlatformAuthenticatorAvailable(); } catch { return false; }
}

async function requestBiometricAuth(): Promise<boolean> {
  try {
    if (typeof window !== 'undefined' && (window as any).ReactNativeWebView) {
      return await requestNativeBiometric();
    }
    return await requestWebAuthn();

}

async function requestNativeBiometric(): Promise<boolean> {
  return new Promise((resolve) => {
    (window as any).ReactNativeWebView?.postMessage(JSON.stringify({ type: 'REQUEST_BIOMETRIC' }));
    const handleMessage = (event: MessageEvent) => {
      try {
        const data = JSON.parse(event.data);
        if (data.type === 'BIOMETRIC_RESULT') {
          window.removeEventListener('message', handleMessage);
          resolve(data.success);
        }
      } catch { }
    };
    window.addEventListener('message', handleMessage);
    setTimeout(() => { window.removeEventListener('message', handleMessage); resolve(false); }, 30000);
  });
}

async function requestWebAuthn(): Promise<boolean> {
  try {
    if (!window.PublicKeyCredential) return await requestPasscode();
    const credentialId = localStorage.getItem('steinz_webauthn_id');
    if (!credentialId) return await requestPasscode();
    const challenge = new Uint8Array(32);
    window.crypto.getRandomValues(challenge);
    const assertion = await navigator.credentials.get({
      publicKey: { challenge, allowCredentials: [{ id: Uint8Array.from(atob(credentialId), c => c.charCodeAt(0)), type: 'public-key' }], timeout: 60000, userVerification: 'required' }
    });
    return assertion !== null;
  } catch { return await requestPasscode(); }
}

async function requestPasscode(): Promise<boolean> {
  return new Promise((resolve) => {
    const passcode = prompt('Enter your 6-digit passcode:');
    if (!passcode) { resolve(false); return; }
    const stored = localStorage.getItem('steinz_passcode');
    resolve(passcode === stored);
  });
}

async function setupWebAuthn(): Promise<boolean> {
  try {
    const challenge = new Uint8Array(32);
    window.crypto.getRandomValues(challenge);
    const userId = new Uint8Array(16);
    window.crypto.getRandomValues(userId);
    const credential = await navigator.credentials.create({
      publicKey: { challenge, rp: { name: 'STEINZ Labs', id: window.location.hostname }, user: { id: userId, name: 'user@steinzlabs.com', displayName: 'STEINZ User' }, pubKeyCredParams: [{ alg: -7, type: 'public-key' }, { alg: -257, type: 'public-key' }], authenticatorSelection: { authenticatorAttachment: 'platform', userVerification: 'required' }, timeout: 60000 }
    });
    if (credential) {
      const credentialId = btoa(String.fromCharCode(...new Uint8Array((credential as any).rawId)));
      localStorage.setItem('steinz_webauthn_id', credentialId);
      return true;
    }
    return false;
  } catch { return false; }
}

export function setupPasscode(): boolean {
  const passcode = prompt('Create a 6-digit passcode:');
  if (!passcode || passcode.length !== 6 || !/^\d+$/.test(passcode)) { alert('Please enter a valid 6-digit passcode'); return false; }
  const confirm = prompt('Confirm your passcode:');
  if (passcode !== confirm) { alert('Passcodes do not match'); return false; }
  localStorage.setItem('steinz_passcode', passcode);
  return true;
}

import { arkhamAPI } from '../arkham/api';

export interface VTXContext {
  userWallet?: string;
  currentToken?: string;
  recentSearches: string[];
  conversationHistory: Array<{ role: 'user' | 'assistant'; content: string }>;
}

export async function processVTXQuery(query: string, context: VTXContext): Promise<string> {
  const lowerQuery = query.toLowerCase();

  // Detect intent
  if (lowerQuery.includes('scam') || lowerQuery.includes('safe') || lowerQuery.includes('risk')) {
    return await handleSecurityQuery(query, context);
  }
  if (lowerQuery.includes('wallet') && (lowerQuery.includes('analyze') || lowerQuery.includes('check'))) {
    return await handleWalletAnalysis(query, context);
  }
  if (lowerQuery.includes('smart money') || lowerQuery.includes('institution') || lowerQuery.includes('whale')) {
    return await handleSmartMoneyQuery(query, context);
  }
  if (lowerQuery.includes('buy') || lowerQuery.includes('trade') || lowerQuery.includes('should i')) {
    return await handleTradingAdvice(query, context);
  }
  if (lowerQuery.includes('token') || lowerQuery.includes('coin')) {
    return await handleTokenQuery(query, context);
  }

  // Fallback: general conversational response
  return await handleGeneralQuery(query, context);
}

async function handleSecurityQuery(query: string, context: VTXContext): Promise<string> {
  // Extract token address from query
  const addressMatch = query.match(/[A-Za-z0-9]{32,44}/);
  if (!addressMatch) {
    return "Please provide a token address and I'll run a complete security analysis using Arkham Intelligence.";
  }

  const tokenAddress = addressMatch[0];

  try {
    // Get holders and check for scammers
    const holders = await arkhamAPI.getTokenHolders(tokenAddress, 20);
    const scammers = holders.filter(h => h.labels?.includes('scammer') || h.labels?.includes('rug_puller'));
    const verified = holders.filter(h => h.entity?.verified);

    let response = `🛡️ **Security Analysis Complete**\n\n`;

    if (scammers.length > 0) {
      response += `⚠️ **WARNING: ${scammers.length} known scammer(s) detected in top holders**\n`;
      response += `Shadow Guardian would BLOCK this trade.\n\n`;
      scammers.slice(0, 3).forEach(s => {
        response += `- ${s.entity?.name || s.address.slice(0, 8)}... holds ${s.percentage.toFixed(2)}%\n`;
      });
      response += `\n❌ **Recommendation: DO NOT BUY**`;
    } else if (verified.length >= 2) {
      response += `✅ **SAFE: ${verified.length} verified institution(s) holding**\n\n`;
      verified.slice(0, 3).forEach(v => {
        response += `- ${v.entity?.name} (${v.entity?.type}) holds ${v.percentage.toFixed(2)}%\n`;
      });
      response += `\n✅ **Recommendation: Low risk, verified entities present**`;
    } else {
      response += `⚠️ **CAUTION: No scammers detected, but no verified institutions either**\n`;
      response += `This is a newer/smaller token. Proceed with caution.\n\n`;
      response += `ℹ️ **Recommendation: Start with small position**`;
    }

    return response;
  } catch (error) {
    return "I encountered an error analyzing this token. Please try again or provide a valid Solana token address.";
  }
}

'use client';
import { useState, useEffect } from 'react';
import { getTokenMetadata } from '@/lib/data/tokenMetadata';

interface TokenLogoProps {
  address: string;
  chain?: string;
  size?: number;
  showSymbol?: boolean;
  className?: string;
}

export function TokenLogo({ address, chain = 'solana', size = 32, showSymbol = false, className = '' }: TokenLogoProps) {
  const [metadata, setMetadata] = useState<any>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadMetadata();
  }, [address, chain]);

  async function loadMetadata() {
    try {
      setLoading(true);
      const meta = await getTokenMetadata(address, chain);
      setMetadata(meta);
    } catch (error) {
      console.error('Failed to load token metadata:', error);
    } finally {
      setLoading(false);
    }
  }

  if (loading) {
    return <div className={`bg-gray-700 rounded-full animate-pulse ${className}`} style={{ width: size, height: size }} />;
  }

  return (
    <div className={`flex items-center gap-2 ${className}`}>
      {metadata?.logo ? (
        <img src={metadata.logo} alt={metadata.symbol} className="rounded-full" style={{ width: size, height: size }} onError={(e) => { (e.target as HTMLImageElement).src = '/placeholder-token.png'; }} />
      ) : (
        <div className="bg-gradient-to-br from-[#0A1EFF] to-purple-500 rounded-full flex items-center justify-center text-white font-bold" style={{ width: size, height: size, fontSize: size / 3 }}>
          {metadata?.symbol?.slice(0, 2) || '?'}
        </div>
      )}
      {showSymbol && metadata && (
        <div>
          <div className="font-medium text-white">{metadata.symbol}</div>
          <div className="text-xs text-gray-500">{metadata.name}</div>
        </div>
      )}
    </div>
  );
}


PART 6: BEAUTIFUL BUBBLEMAPS (3D EFFECTS)
File: components/visualization/Bubblemaps.tsx

'use client';
import { useEffect, useRef, useState } from 'react';
import * as d3 from 'd3';

interface BubbleNode {
  id: string;
  label: string;
  percentage: number;
  valueUSD: string;
  color: string;
  size: number;
  verified: boolean;
  isScammer: boolean;
  x?: number;
  y?: number;
}

export function Bubblemaps({ nodes, width = 1000, height = 600, onNodeClick }: { nodes: BubbleNode[]; width?: number; height?: number; onNodeClick?: (node: BubbleNode) => void }) {
  const svgRef = useRef<SVGSVGElement>(null);
  const [hovered, setHovered] = useState<BubbleNode | null>(null);

  useEffect(() => {
    if (!svgRef.current || !nodes.length) return;

    d3.select(svgRef.current).selectAll('*').remove();

    const svg = d3.select(svgRef.current).attr('width', width).attr('height', height).style('background', '#0A0E1A').style('border-radius', '12px');

    const defs = svg.append('defs');
    const filter = defs.append('filter').attr('id', 'glow').attr('x', '-50%').attr('y', '-50%').attr('width', '200%').attr('height', '200%');
    filter.append('feGaussianBlur').attr('stdDeviation', '4').attr('result', 'coloredBlur');
    const feMerge = filter.append('feMerge');
    feMerge.append('feMergeNode').attr('in', 'coloredBlur');
    feMerge.append('feMergeNode').attr('in', 'SourceGraphic');

    const gradient = defs.append('radialGradient').attr('id', 'verified-gradient');
    gradient.append('stop').attr('offset', '0%').attr('stop-color', '#00FF88').attr('stop-opacity', 0.8);
    gradient.append('stop').attr('offset', '100%').attr('stop-color', '#00FF88').attr('stop-opacity', 0.3);

    const sizeScale = d3.scalePow().exponent(0.5).domain([0, d3.max(nodes, d => d.percentage) || 100]).range([30, 150]);
    const data = nodes.map(n => ({ ...n, radius: sizeScale(n.percentage) }));

    const simulation = d3.forceSimulation(data as any).force('charge', d3.forceManyBody().strength(d => -Math.pow((d as any).radius, 2) * 0.5)).force('center', d3.forceCenter(width / 2, height / 2)).force('collision', d3.forceCollide().radius((d: any) => d.radius + 10).strength(0.9)).force('x', d3.forceX(width / 2).strength(0.05)).force('y', d3.forceY(height / 2).strength(0.05));

    const container = svg.append('g').attr('class', 'bubbles');
    const nodeGroups = container.selectAll('g').data(data).enter().append('g').attr('class', 'bubble-node').style('cursor', 'pointer').on('click', (event, d: any) => { if (onNodeClick) onNodeClick(d); }).on('mouseenter', (event, d: any) => { setHovered(d); d3.select(event.currentTarget).transition().duration(200).attr('transform', 'scale(1.1)'); }).on('mouseleave', (event) => { setHovered(null); d3.select(event.currentTarget).transition().duration(200).attr('transform', 'scale(1)'); });

    nodeGroups.each(function(d: any) {
      const group = d3.select(this);
      if (d.verified || d.isScammer) {
        group.append('circle').attr('r', d.radius + 4).attr('fill', 'none').attr('stroke', d.verified ? '#00FF88' : '#FF4444').attr('stroke-width', 3).attr('opacity', 0.8).style('filter', 'url(#glow)');
      }
      group.append('circle').attr('r', d.radius).attr('fill', d.verified ? 'url(#verified-gradient)' : d.color).attr('opacity', 0.9).attr('stroke', d.isScammer ? '#FF4444' : 'rgba(255,255,255,0.2)').attr('stroke-width', 2);
      group.append('circle').attr('r', d.radius * 0.4).attr('cx', -d.radius * 0.2).attr('cy', -d.radius * 0.2).attr('fill', 'rgba(255,255,255,0.3)').attr('opacity', 0.5);
    });

    nodeGroups.append('text').text((d: any) => d.label).attr('text-anchor', 'middle').attr('dy', '0.3em').attr('font-size', (d: any) => Math.min(d.radius * 0.25, 14)).attr('font-weight', 'bold').attr('fill', '#FFFFFF').attr('pointer-events', 'none').style('text-shadow', '0 0 4px rgba(0,0,0,0.8)');

    nodeGroups.append('text').text((d: any) => `${d.percentage.toFixed(1)}%`).attr('text-anchor', 'middle').attr('dy', (d: any) => d.radius > 60 ? '2em' : '1.8em').attr('font-size', (d: any) => Math.min(d.radius * 0.2, 12)).attr('fill', '#CCCCCC').attr('pointer-events', 'none').style('text-shadow', '0 0 4px rgba(0,0,0,0.8)');

    simulation.on('tick', () => {
      nodeGroups.attr('transform', (d: any) => `translate(${d.x},${d.y})`);
    });

    return () => { simulation.stop(); };
  }, [nodes, width, height, onNodeClick]);

  return (
    <div className="relative">
      <svg ref={svgRef} className="w-full h-auto" />
      {hovered && (
        <div className="absolute bottom-4 left-4 bg-[#141824] border border-[#1E2433] rounded-lg p-4 max-w-xs z-10">
          <div className="font-bold text-white mb-2">{hovered.label}</div>
          <div className="grid grid-cols-2 gap-2 text-xs">
            <div><div className="text-gray-500">Holdings</div><div className="text-white font-mono">{hovered.percentage.toFixed(2)}%</div></div>
            <div><div className="text-gray-500">Value</div><div className="text-white font-mono">${hovered.valueUSD}</div></div>
          </div>
        </div>
      )}
    </div>
  );
}
