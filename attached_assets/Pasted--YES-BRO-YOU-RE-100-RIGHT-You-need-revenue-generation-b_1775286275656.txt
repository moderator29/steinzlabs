import { supabaseAdmin } from '../database/supabase';

export interface FeeConfiguration {
  swapFeeBps: number;           // Basis points (50 = 0.5%)
  copyTradeFeeBps: number;      // Fee for copy trades
  limitOrderFeeBps: number;     // Fee for limit orders
  dcaFeeBps: number;            // Fee for DCA buys
  treasuryWallet: string;       // Your wallet address
}

export interface FeeCalculation {
  tradeAmount: string;          // Original trade amount
  feeAmount: string;            // Fee to collect
  netAmount: string;            // Amount after fee
  feeBps: number;
  treasuryWallet: string;
}

export interface RevenueRecord {

const FEE_CONFIG: FeeConfiguration = {
  swapFeeBps: 50,              // 0.5% on normal swaps
  copyTradeFeeBps: 100,        // 1% on copy trades (higher value service)
  limitOrderFeeBps: 50,        // 0.5% on limit orders
  dcaFeeBps: 50,               // 0.5% on DCA buys
  treasuryWallet: process.env.TREASURY_WALLET_ADDRESS!, // YOUR WALLET
};

// =====================================================================
// Calculate fee for a trade
// =====================================================================
export function calculateFee(
  tradeAmount: number,
  tradeType: 'SWAP' | 'COPY_TRADE' | 'LIMIT_ORDER' | 'DCA'
): FeeCalculation {
  let feeBps: number;

  switch (tradeType) {
    case 'SWAP':
      feeBps = FEE_CONFIG.swapFeeBps;
      break;
    case 'COPY_TRADE':
      feeBps = FEE_CONFIG.copyTradeFeeBps;
      break;
    case 'LIMIT_ORDER':
      feeBps = FEE_CONFIG.limitOrderFeeBps;
      break;
    case 'DCA':
      feeBps = FEE_CONFIG.dcaFeeBps;
      break;
    default:
      feeBps = FEE_CONFIG.swapFeeBps;
  }

  // Calculate fee (basis points to percentage)
  const feePercentage = feeBps / 10000;
  const feeAmount = tradeAmount * feePercentage;
  const netAmount = tradeAmount - feeAmount;

  return {
    tradeAmount: tradeAmount.toString(),
    feeAmount: feeAmount.toString(),
    netAmount: netAmount.toString(),
    feeBps,
    treasuryWallet: FEE_CONFIG.treasuryWallet,
  };
}

// Get total revenue (analytics)

export async function getTotalRevenue(timeframe?: {
  startDate: string;
  endDate: string;
}): Promise<{
  totalRevenue: number;
  revenueByType: Record<string, number>;
  totalTrades: number;
}> {
  try {
    let query = supabaseAdmin
      .from('revenue_records')
      .select('*')
      .eq('status', 'COLLECTED');

    if (timeframe) {
      query = query
        .gte('timestamp', timeframe.startDate)
        .lte('timestamp', timeframe.endDate);
    }

    const { data: records } = await query;

    if (!records || records.length === 0) {
      return {
        totalRevenue: 0,
        revenueByType: {},
        totalTrades: 0,
      };
    }

    const totalRevenue = records.reduce(
      (sum, r) => sum + parseFloat(r.fee_amount),
      0
    );

    const revenueByType: Record<string, number> = {};
    for (const record of records) {
      const type = record.trade_type;
      if (!revenueByType[type]) {
        revenueByType[type] = 0;
      }
      revenueByType[type] += parseFloat(record.fee_amount);
    }

    return {
      totalRevenue,
      revenueByType,
      totalTrades: records.length,
    };
  } catch (error) {
    console.error('Failed to get total revenue:', error);
    return {
      totalRevenue: 0,
      revenueByType: {},
      totalTrades: 0,
    };
  }
}

);

import { NextRequest, NextResponse } from 'next/server';
import { getUserFeesPaid } from '@/lib/revenue/feeSystem';

export async function GET(request: NextRequest) {
  try {
    const wallet = request.nextUrl.searchParams.get('wallet');

    if (!wallet) {
      return NextResponse.json(
        { error: 'wallet parameter required' },
        { status: 400 }
      );
    }

    const user = await getUserByWallet(wallet);
    if (!user) {
      return NextResponse.json(
        { error: 'User not found' },
        { status: 404 }
      );
    }

      { error: error.message || 'Failed to get fees paid' },
      { status: 500 }
    );
  }
